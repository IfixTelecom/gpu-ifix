    package auditctx

    import "context"

    type upstreamOverrideKey struct{}

    // WithUpstreamOverride attaches an upstream name (e.g., "openrouter-chat")
    // to ctx. The dispatcher (Phase 3) reads this via UpstreamOverrideFromContext
    // and forces the corresponding upstream regardless of tier-0 breaker state.
    func WithUpstreamOverride(ctx context.Context, upstreamName string) context.Context {
        return context.WithValue(ctx, upstreamOverrideKey{}, upstreamName)
    }

    // UpstreamOverrideFromContext returns the upstream name set by
    // WithUpstreamOverride, or "" if none.
    func UpstreamOverrideFromContext(ctx context.Context) string {
        v, _ := ctx.Value(upstreamOverrideKey{}).(string)
        return v
    }

    package quota

    import (
        "context"
        "errors"
        "log/slog"
        "net/http"
        "strconv"
        "time"

        "github.com/google/uuid"
        "github.com/redis/go-redis/v9"

        "github.com/ifixtelecom/gpu-ifix/gateway/internal/auth"
        "github.com/ifixtelecom/gpu-ifix/gateway/internal/httpx"
        "github.com/ifixtelecom/gpu-ifix/gateway/internal/idempotency"
        "github.com/ifixtelecom/gpu-ifix/gateway/internal/obs"
        "github.com/ifixtelecom/gpu-ifix/gateway/internal/tenants"
    )

    // RateLimitMiddleware enforces RPS+RPM atomic via Redis Lua. Per D-D1, this
    // runs BEFORE quotaMiddleware. Per D-A2 it is fail-open: if the Lua call
    // errors AND cfg.RateLimitFailOpen is true, the request passes; the metric
    // GatewayRateLimitCheckFailures is incremented.
    //
    // Per D-D1 the rate-limit check is SKIPPED on idempotency replays (replay
    // already consumed the bucket on the first attempt).
    func RateLimitMiddleware(
        rdb redis.UniversalClient,
        loader *tenants.Loader,
        failOpen bool,
        log *slog.Logger,
    ) func(http.Handler) http.Handler {
        log = log.With("module", "QUOTA")
        return func(next http.Handler) http.Handler {
            return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
                if idempotency.IsReplay(r.Context()) {
                    next.ServeHTTP(w, r)
                    return
                }
                ac, ok := auth.FromContext(r.Context())
                if !ok {
                    httpx.WriteOpenAIError(w, http.StatusUnauthorized,
                        "authentication_error", "no_api_key",
                        "Authenticated tenant required.")
                    return
                }
                cfg, err := loader.Get(ac.TenantID)
                if err != nil {
                    // Tenant unknown — fall through, auth would have caught it.
                    next.ServeHTTP(w, r)
                    return
                }
                routeClass := classifyRoute(r.URL.Path)
                bucketCfg := BucketConfig{RPSCapacity: cfg.RPSLimit, RPMCapacity: cfg.RPMLimit}
                res, err := CheckBuckets(r.Context(), rdb,
                    ac.TenantID.String(), string(routeClass),
                    bucketCfg.RPSCapacity, bucketCfg.RPMCapacity,
                    bucketCfg.RPSRefillPerMs(), bucketCfg.RPMRefillPerMs(),
                    1, time.Now().UnixMilli(),
                )
                if err != nil {
                    obs.GatewayRateLimitCheckFailures.WithLabelValues("transport").Inc()
                    if failOpen {
                        log.Warn("rate-limit Lua error; failing open", "tenant", cfg.Slug, "err", err)
                        next.ServeHTTP(w, r)
                        return
                    }
                    httpx.WriteOpenAIError(w, http.StatusServiceUnavailable,
                        "service_unavailable", "rate_limit_check_unavailable",
                        "Rate-limit check unavailable.")
                    return
                }
                // Always set rate-limit headers regardless of allowed (OpenAI-compat).
                w.Header().Set("X-RateLimit-Limit-Requests", strconv.Itoa(bucketCfg.RPSCapacity))
                w.Header().Set("X-RateLimit-Remaining-Requests", strconv.Itoa(res.RemRPS))
                if !res.Allowed {
                    obs.GatewayRateLimitRejected.WithLabelValues(cfg.Slug, res.FailedWindow).Inc()
                    var code, msg string
                    var retryMs int
                    if res.FailedWindow == "rps" {
                        code = "rate_limit_rps"
                        msg = "Rate limit exceeded: requests per second."
                        retryMs = res.ResetRPSms
                    } else {
                        code = "rate_limit_rpm"
                        msg = "Rate limit exceeded: requests per minute."
                        retryMs = res.ResetRPMms
                    }
                    if retryMs < 1000 { retryMs = 1000 }
                    w.Header().Set("Retry-After", strconv.Itoa(retryMs/1000))
                    httpx.WriteOpenAIError(w, http.StatusTooManyRequests,
                        "rate_limit_error", code, msg)
                    return
                }
                next.ServeHTTP(w, r)
            })
        }
    }

    // QuotaMiddleware enforces daily + monthly quotas. Per D-A2 it is fail-closed:
    // any DB error returns 503 quota_check_unavailable (refusing to risk runaway
    // external cost on undefined quota state).
    //
    // Per D-D1 quota CHECK runs even on idempotency replays (replay still consumes
    // quota — Stripe semantics).
    func QuotaMiddleware(
        checker *QuotaChecker,
        loader *tenants.Loader,
        log *slog.Logger,
    ) func(http.Handler) http.Handler {
        log = log.With("module", "QUOTA")
        return func(next http.Handler) http.Handler {
            return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
                ac, ok := auth.FromContext(r.Context())
                if !ok {
                    next.ServeHTTP(w, r) // already handled by RateLimit
                    return
                }
                cfg, err := loader.Get(ac.TenantID)
                if err != nil {
                    next.ServeHTTP(w, r)
                    return
                }
                limits := QuotaLimits{
                    DailyTokens:         cfg.DailyQuotaTokens,
                    MonthlyTokens:       cfg.MonthlyQuotaTokens,
                    DailyAudioMinutes:   cfg.DailyQuotaAudioMinutes,
                    MonthlyAudioMinutes: cfg.MonthlyQuotaAudioMinutes,
                    DailyEmbeds:         cfg.DailyQuotaEmbeds,
                    MonthlyEmbeds:       cfg.MonthlyQuotaEmbeds,
                }
                if err := checker.CheckQuotaToday(r.Context(), ac.TenantID, limits); err != nil {
                    handleQuotaError(w, cfg.Slug, err, "daily")
                    return
                }
                if err := checker.CheckQuotaMonth(r.Context(), ac.TenantID, limits); err != nil {
                    handleQuotaError(w, cfg.Slug, err, "monthly")
                    return
                }
                next.ServeHTTP(w, r)
            })
        }
    }

    func handleQuotaError(w http.ResponseWriter, tenantSlug string, err error, period string) {
        if errors.Is(err, ErrQuotaCheckUnavailable) {
            httpx.WriteOpenAIError(w, http.StatusServiceUnavailable,
                "service_unavailable", "quota_check_unavailable",
                "Quota check unavailable; refusing to risk runaway external cost.")
            return
        }
        code := ErrorCode(err)
        var dimension string
        switch {
        case errors.Is(err, ErrQuotaExceededDailyTokens), errors.Is(err, ErrQuotaExceededMonthlyTokens):
            dimension = "tokens"
        case errors.Is(err, ErrQuotaExceededDailyAudioMinutes), errors.Is(err, ErrQuotaExceededMonthlyAudioMinutes):
            dimension = "audio_minutes"
        case errors.Is(err, ErrQuotaExceededDailyEmbeds), errors.Is(err, ErrQuotaExceededMonthlyEmbeds):
            dimension = "embeds"
        }
        obs.GatewayQuotaRejected.WithLabelValues(tenantSlug, dimension, period).Inc()
        httpx.WriteOpenAIError(w, http.StatusTooManyRequests,
            "insufficient_quota", code, "Quota exceeded.")
    }

    // classifyRoute maps a chi route path to a RouteClass for namespacing
    // rate-limit buckets. Falls back to chat for unknown paths.
    func classifyRoute(path string) RouteClass {
        switch {
        case path == "/v1/chat/completions":
            return RouteClassChat
        case path == "/v1/embeddings":
            return RouteClassEmbed
        case path == "/v1/audio/transcriptions":
            return RouteClassSTT
        }
        return RouteClassChat
    }

    // helper for tests/main to verify tenant id parsing
    func parseTenantID(s string) (uuid.UUID, error) {
        return uuid.Parse(s)
    }

    var _ = context.Background // silence unused import on some builds
    ```

    package schedule

    import (
        "log/slog"
        "net/http"
        "time"

        "github.com/ifixtelecom/gpu-ifix/gateway/internal/auditctx"
        "github.com/ifixtelecom/gpu-ifix/gateway/internal/auth"
        "github.com/ifixtelecom/gpu-ifix/gateway/internal/obs"
        "github.com/ifixtelecom/gpu-ifix/gateway/internal/tenants"
    )

    // upstreamForTier maps tier int → upstream name expected by the dispatcher.
    // Today: tier 0 → local-llm (default; no override needed); tier 1 → openrouter-chat.
    func upstreamForTier(tier int) string {
        if tier == Tier1 {
            return "openrouter-chat"
        }
        return ""
    }

    // Middleware decides upstream override based on tenant Mode + clock.
    // Per D-C2: peak off-hours forces tier-1 (openrouter-chat) regardless of
    // tier-0 breaker state. 24/7 mode never overrides.
    //
    // Decision is written to ctx via auditctx.WithUpstreamOverride, which the
    // dispatcher (Plan 04-06 Task 2) reads to bypass tier-0 selection.
    func Middleware(loader *tenants.Loader, log *slog.Logger) func(http.Handler) http.Handler {
        log = log.With("module", "SCHEDULE")
        return func(next http.Handler) http.Handler {
            return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
                ac, ok := auth.FromContext(r.Context())
                if !ok {
                    next.ServeHTTP(w, r)
                    return
                }
                cfg, err := loader.Get(ac.TenantID)
                if err != nil {
                    next.ServeHTTP(w, r)
                    return
                }
                ctx := r.Context()
                tier := DecideUpstreamTier(cfg, time.Now())
                if name := upstreamForTier(tier); name != "" {
                    ctx = auditctx.WithUpstreamOverride(ctx, name)
                    obs.GatewayScheduleRouting.WithLabelValues(cfg.Slug, "off_hours_external").Inc()
                } else {
                    obs.GatewayScheduleRouting.WithLabelValues(cfg.Slug, "local").Inc()
                }
                next.ServeHTTP(w, r.WithContext(ctx))
            })
        }
    }
    ```

    package obs

    import (
        "log/slog"
        "net/http"
        "strconv"

        "github.com/go-chi/chi/v5"
    )

    // statusRecorder captures the response status code for label cardinality control.
    type statusRecorder struct {
        http.ResponseWriter
        status int
    }

    func (s *statusRecorder) WriteHeader(code int) {
        s.status = code
        s.ResponseWriter.WriteHeader(code)
    }

    // RequestsMiddleware (folded TODO Phase 4) instruments every request with
    // RequestsTotal{route, status_class} after the response has been written.
    // route uses chi's RoutePattern (e.g. "/v1/chat/completions") to keep
    // cardinality bounded; status_class is "2xx"/"3xx"/"4xx"/"5xx".
    func RequestsMiddleware(_ *slog.Logger) func(http.Handler) http.Handler {
        return func(next http.Handler) http.Handler {
            return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
                rec := &statusRecorder{ResponseWriter: w, status: http.StatusOK}
                next.ServeHTTP(rec, r)
