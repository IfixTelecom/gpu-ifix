Wave 4 — admin surface. Closes the operational loop: every Phase 4 mutation that admin staff need (tenant mode, tenant quota, prices, fx, admin keys, reconcile) gets a `gatewayctl` subcommand. The reconcile command implements the drift detection from D-B1: compares the cached `usage_counters` against the authoritative SUM(billing_events) and optionally rewrites the cache.

Per CONTEXT.md D-D4 (full subcommand inventory) and D-C1 (gatewayctl rejects sensitive+peak pre-DB before the CHECK constraint fires) and D-B3 (price hot-reload via gatewayctl + NOTIFY).

Purpose: Ops cannot manage Phase 4 features (per-tenant quotas, peak windows, price changes, admin key rotation) without a CLI. The Phase 2 D-A3 pattern established `gatewayctl` as the admin surface; Phase 4 expands it to cover the new domains. Without this, runtime configuration of Phase 4 requires manual SQL (error-prone, no audit trail).

Output: 5 new/extended subcommand files under `gateway/cmd/gatewayctl/`, a dispatcher update in main.go, and unit tests for the validation paths (sensitive+peak rejection, partial-UPDATE flag handling, drift detection).

    func runTenantSetMode(ctx context.Context, args []string, log *slog.Logger) int {
        fs := flag.NewFlagSet("tenant set-mode", flag.ExitOnError)
        slug := fs.String("tenant", "", "tenant slug (required)")
        mode := fs.String("mode", "", "24/7 | peak (required)")
        window := fs.String("window", "", "HH-HH (e.g. 08-22) only with --mode peak")
        tz := fs.String("tz", "America/Sao_Paulo", "IANA timezone")
        if err := fs.Parse(args); err != nil { return 2 }
        if *slug == "" || *mode == "" {
            fs.Usage(); return 2
        }
        if *mode != "24/7" && *mode != "peak" {
            fmt.Fprintf(os.Stderr, "invalid --mode %q; must be 24/7 or peak\n", *mode); return 2
        }

        _, pool, err := loadAndPool(ctx, log)
        if err != nil { return 1 }
        defer pool.Close()
        q := gen.New(pool)

        // D-C1 path 1: pre-DB validation — reject sensitive + peak before SQL CHECK fires
        if *mode == "peak" {
            tenant, err := q.GetTenantConfigBySlug(ctx, *slug) // Add this query if not in tenants_admin.sql
            if err != nil {
                fmt.Fprintf(os.Stderr, "tenant %q not found: %v\n", *slug, err); return 1
            }
            if tenant.DataClass == "sensitive" {
                fmt.Fprintf(os.Stderr,
                    "cannot set peak mode for sensitive tenant %q (LGPD policy: external providers blocked)\n",
                    *slug)
                return 2
            }
        }

        var startT, endT pgtype.Time
        if *mode == "peak" {
            startH, endH, perr := parseWindowHours(*window)
            if perr != nil {
                fmt.Fprintln(os.Stderr, perr); return 2
            }
            // pgtype.Time stores microseconds since midnight
            _ = startT.Scan(fmt.Sprintf("%02d:00:00", startH))
            _ = endT.Scan(fmt.Sprintf("%02d:00:00", endH))
        }

        if err := q.UpdateTenantMode(ctx, gen.UpdateTenantModeParams{
            Slug: *slug,
            Mode: *mode,
            PeakWindowStart: startT,
            PeakWindowEnd:   endT,
            ScheduleTimezone: pgtype.Text{String: *tz, Valid: *tz != ""},
        }); err != nil {
            fmt.Fprintf(os.Stderr, "update failed: %v\n", err); return 1
        }
        log.Info("tenant mode updated", "slug", *slug, "mode", *mode)
        return 0
    }

    func parseWindowHours(s string) (start, end int, err error) {
        var s1, s2 int
        n, err := fmt.Sscanf(s, "%d-%d", &s1, &s2)
        if err != nil || n != 2 || s1 < 0 || s1 > 23 || s2 < 0 || s2 > 23 {
            return 0, 0, fmt.Errorf("invalid --window %q; expected HH-HH (e.g. 08-22)", s)
        }
        return s1, s2, nil
    }

    func runPrices(ctx context.Context, args []string, log *slog.Logger) int {
        if len(args) == 0 {
            fmt.Fprintln(os.Stderr, "Usage: gatewayctl prices <set|list|set-fx> [flags]")
            return 2
        }
        switch args[0] {
        case "set":     return runPricesSet(ctx, args[1:], log)
        case "list":    return runPricesList(ctx, args[1:], log)
        case "set-fx":  return runPricesSetFX(ctx, args[1:], log)
        default:
            fmt.Fprintf(os.Stderr, "unknown subcommand: %s\n", args[0])
            return 2
        }

    func runPricesSet(ctx context.Context, args []string, log *slog.Logger) int {
        fs := flag.NewFlagSet("prices set", flag.ExitOnError)
        model := fs.String("model", "", "model name (required)")
        provider := fs.String("provider", "", "provider (required)")
        unit := fs.String("unit", "", "input_token|output_token|audio_second|embed_request (required)")
        usd := fs.Float64("usd", 0, "unit cost USD (required, > 0)")
        notes := fs.String("notes", "", "free-form note")
        if err := fs.Parse(args); err != nil { return 2 }
        if *model == "" || *provider == "" || *unit == "" || *usd <= 0 {
            fs.Usage(); return 2
        }

        _, pool, err := loadAndPool(ctx, log)
        if err != nil { return 1 }
        defer pool.Close()

        // Atomic swap in a single txn so the trigger fires NOTIFY exactly once.
        tx, err := pool.BeginTx(ctx, pgx.TxOptions{})
        if err != nil { return 1 }
        defer func() { _ = tx.Rollback(ctx) }()
        q := gen.New(tx)
        if err := q.ExpireActivePrice(ctx, gen.ExpireActivePriceParams{
            Model: *model, Provider: *provider, Unit: *unit,
        }); err != nil {
            fmt.Fprintf(os.Stderr, "expire failed: %v\n", err); return 1
        }
        var notesParam pgtype.Text
        if *notes != "" { notesParam = pgtype.Text{String: *notes, Valid: true} }
        var usdNum pgtype.Numeric
        _ = usdNum.Scan(fmt.Sprintf("%.8f", *usd))
        ins, err := q.InsertPrice(ctx, gen.InsertPriceParams{
            Model: *model, Provider: *provider, Unit: *unit,
            UnitCostUsd: usdNum,
            Notes: notesParam,
        })
        if err != nil {
            fmt.Fprintf(os.Stderr, "insert failed: %v\n", err); return 1
        }
        if err := tx.Commit(ctx); err != nil {
            fmt.Fprintf(os.Stderr, "commit failed: %v\n", err); return 1
        }
        log.Info("price updated", "model", *model, "provider", *provider, "unit", *unit, "usd", *usd, "id", ins.ID)
        return 0
    }

        }
        _ = w.Flush()
        return 0

    func runAdminKey(ctx context.Context, args []string, log *slog.Logger) int {

    func runAdminKeyCreate(ctx context.Context, args []string, log *slog.Logger) int {
        fs := flag.NewFlagSet("admin-key create", flag.ExitOnError)
        label := fs.String("label", "", "human label (required)")
        if err := fs.Parse(args); err != nil { return 2 }
        if *label == "" { fs.Usage(); return 2 }

        _, pool, err := loadAndPool(ctx, log)
        if err != nil { return 1 }
        defer pool.Close()
        q := gen.New(pool)

        // Generate raw key
        buf := make([]byte, 16)
        if _, err := rand.Read(buf); err != nil {
            fmt.Fprintln(os.Stderr, "rand: ", err); return 1
        }
        raw := "ifix_admin_" + hex.EncodeToString(buf)
        sum := sha256.Sum256([]byte(raw))
        bcryptHash, err := bcrypt.GenerateFromPassword([]byte(raw), 10)
        if err != nil {
            fmt.Fprintln(os.Stderr, "bcrypt: ", err); return 1
        }
        prefix := "ifix_admin_****" + raw[len(raw)-4:]
        ins, err := q.InsertAdminKey(ctx, gen.InsertAdminKeyParams{
            KeyLookupHash: sum[:],
            KeyHash:       string(bcryptHash),
            KeyPrefix:     prefix,
            Label:         *label,
        })
        if err != nil {
            fmt.Fprintln(os.Stderr, "insert: ", err); return 1
