    import (
        "context"

    // SC-5: 1000 concurrent goroutines vs RPS=100 → exactly 100 pass.
    // The atomicity is guaranteed by Redis Lua's single-threaded semantics; this
    // test verifies the Go wrapper preserves it.
    func TestRateLimitAtomic1000Concurrent(t *testing.T) {
        ctx := context.Background()
        freshSchema(t)
        seed := seedPhase4(t, ctx)
        _ = seed

        // Set tenant rps=100 via direct SQL
        if _, err := dbPool.Exec(ctx,
            `UPDATE ai_gateway.tenants SET rps_limit = 100 WHERE id = $1`,
            seed.ConverseAITenantID); err != nil {
            t.Fatalf("set rps: %v", err)
        }

        log := slog.New(slog.NewJSONHandler(io.Discard, nil))
        loc, _ := time.LoadLocation("America/Sao_Paulo")
        loader, err := tenants.NewLoader(ctx, dbPool, loc, log)
        if err != nil { t.Fatal(err) }

        // Use the live middleware over a stub upstream
        upstream := http.HandlerFunc(func(w http.ResponseWriter, _ *http.Request) {
            w.WriteHeader(http.StatusOK)
        })
        // Inject a fake auth context so RateLimitMiddleware finds the tenant
        h := injectAuthContext(seed.ConverseAITenantID,
            quota.RateLimitMiddleware(redisClient, loader, false, log)(upstream))

        var allowed, denied atomic.Uint32
        wg := sync.WaitGroup{}
        const N = 1000
        for i := 0; i < N; i++ {
            wg.Add(1)
            go func() {
                defer wg.Done()
                rec := httptest.NewRecorder()
                req := httptest.NewRequest("POST", "/v1/chat/completions", nil)
                h.ServeHTTP(rec, req)
                if rec.Code == http.StatusOK {
                    allowed.Add(1)
                } else if rec.Code == http.StatusTooManyRequests {
                    denied.Add(1)
                }
            }()
        }
        wg.Wait()

        if a := allowed.Load(); a != 100 {
            t.Errorf("Lua atomic: want exactly 100 allowed, got %d (drift = %d)",
                a, int32(a)-100)
        }
        if d := denied.Load(); d != 900 {
            t.Errorf("expected 900 denied (429), got %d", d)
        }
        if total := allowed.Load() + denied.Load(); total != N {
            t.Errorf("total != N: got %d (some requests neither allowed nor 429)", total)
        }
    }

    // helper: injects a synthetic auth.Context for the test tenant
    func injectAuthContext(tenantID uuid.UUID, next http.Handler) http.Handler {
        // ... use auth.WithContext from gateway/internal/auth/context.go to wrap ctx ...
        // Detail elided here — the executor wires the matching helper from auth package.
        return next
    }

    package integration_test

    import (
        "context"
        "io"
        "log/slog"
        "testing"
        "time"

        "github.com/ifixtelecom/gpu-ifix/gateway/internal/quota"
    )

    // SC-1 (quota part): yesterday's quota does not block today's requests.
    func TestQuotaDailyRolloverBRT(t *testing.T) {
        ctx := context.Background()
        freshSchema(t)
        seed := seedPhase4(t, ctx)

        // Set daily_quota_tokens = 100 for the tenant
        if _, err := dbPool.Exec(ctx,
            `UPDATE ai_gateway.tenants SET daily_quota_tokens = 100 WHERE id = $1`,
            seed.ConverseAITenantID); err != nil {
            t.Fatal(err)
        }

        // Pre-seed yesterday's row with tokens_in=99
        loc, _ := time.LoadLocation("America/Sao_Paulo")
        yesterday := time.Now().In(loc).AddDate(0, 0, -1).Format("2006-01-02")
        if _, err := dbPool.Exec(ctx, `
            INSERT INTO ai_gateway.usage_counters (tenant_id, date, tokens_in, tokens_out, requests_count)
            VALUES ($1, $2, 99, 2, 1)
        `, seed.ConverseAITenantID, yesterday); err != nil {
            t.Fatal(err)
        }

        // Today's row absent → CheckQuotaToday should return nil (no daily lookup row)
        log := slog.New(slog.NewJSONHandler(io.Discard, nil))
        checker := quota.NewQuotaChecker(genQueries(), log)
        err := checker.CheckQuotaToday(ctx, seed.ConverseAITenantID, quota.QuotaLimits{
            DailyTokens: 100, MonthlyTokens: 1_000_000,
            DailyAudioMinutes: 600, MonthlyAudioMinutes: 18000,
            DailyEmbeds: 100_000, MonthlyEmbeds: 3_000_000,
        })
        if err != nil {
            t.Errorf("today's quota should NOT be blocked by yesterday's row, got: %v", err)
        }

        // Now insert TODAY's row at 99+2=101 → over the daily limit of 100
        today := time.Now().In(loc).Format("2006-01-02")
        if _, err := dbPool.Exec(ctx, `
            INSERT INTO ai_gateway.usage_counters (tenant_id, date, tokens_in, tokens_out, requests_count)
            VALUES ($1, $2, 99, 2, 1)
        `, seed.ConverseAITenantID, today); err != nil {
            t.Fatal(err)
        }
        err = checker.CheckQuotaToday(ctx, seed.ConverseAITenantID, quota.QuotaLimits{
            DailyTokens: 100, MonthlyTokens: 1_000_000,
            DailyAudioMinutes: 600, MonthlyAudioMinutes: 18000,
            DailyEmbeds: 100_000, MonthlyEmbeds: 3_000_000,
        })
        if !errors.Is(err, quota.ErrQuotaExceededDailyTokens) {
            t.Errorf("today over limit should return ErrQuotaExceededDailyTokens, got: %v", err)
        }
    }

    func genQueries() *gen.Queries { return gen.New(dbPool) }
