```go
// AFTER Phase 6.6 (per reviews suggestion #13 — mutex over atomic.Pointer):
type Scraper struct {
    urlMu    sync.RWMutex
    url      string
    interval time.Duration
    timeout  time.Duration
    log      *slog.Logger
    // ... other fields
}

func New(url string, ...) *Scraper { return &Scraper{url: url, ...} }

// SetURL replaces the scrape target URL at runtime.
// Phase 6.6 — primary pod is dynamic, so scrape URL must change when primary
// transitions Asleep → Provisioning → Ready and Ready → Asleep.
// Reviews suggestion #13 (Codex LOW): chose sync.RWMutex over atomic.Pointer[string]
// because it's less invasive — existing scrape() reads s.url as a plain field;
// switching to atomic.Pointer would force every read to .Load() everywhere,
// rippling through tests + other call sites. RWMutex preserves the read pattern
// (read lock around s.url access, write lock around SetURL).
func (s *Scraper) SetURL(url string) {
    s.urlMu.Lock()
    defer s.urlMu.Unlock()
    s.url = url
}

func (s *Scraper) scrape() {
    s.urlMu.RLock()
    url := s.url
    s.urlMu.RUnlock()
    if url == "" {
        // Empty URL → fail-open (Phase 5 design preserved)
        return
    }
    // use url for HTTP scrape
}
```
