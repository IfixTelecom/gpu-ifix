<!-- Overnight wrap day-filter semantics (reviews consensus action #5):
     Schedule: UpHour=22, DownHour=8, Days={mon: true, others: false}
     Time: Tuesday 02:00 (BRT)
     Expected: IsInPeak() == true  (Monday 22:00 → Tuesday 08:00 window IS active because Monday's bit is enabled)

     Schedule: UpHour=22, DownHour=8, Days={tue: true, others: false}
     Time: Tuesday 02:00
     Expected: IsInPeak() == false (no Monday bit enabled; Tuesday's own UpHour=22 hasn't fired yet)

     Algorithm:
     if r.Disabled { return false }
     hour := now.In(r.Timezone).Hour()
     weekday := now.In(r.Timezone).Weekday()
     if r.UpHour < r.DownHour {   // simple window mon..fri 08-22
         if !r.Days[weekday] { return false }
         return hour >= r.UpHour && hour < r.DownHour
     }
     // Overnight wrap: window straddles midnight
     // Case A: now is in [UpHour, 24) — uses today's enabled bit
     if hour >= r.UpHour {
         return r.Days[weekday]
     }
     // Case B: now is in [0, DownHour) — uses YESTERDAY's enabled bit
     // (the originator of the wrap)
     if hour < r.DownHour {
         yesterday := (weekday - 1 + 7) % 7
         return r.Days[time.Weekday(yesterday)]
     }
     return false
-->

<!-- Pre-warm offset (reviews consensus action #8):
     ShouldBeProvisioned returns true ProvisionLeadS BEFORE UpHour transition AND throughout active window
     With UpHour=8 + ProvisionLeadS=1800 (default per Plan 06.6-03) → kicks in at 07:30
     Implementation:
     if r.Disabled { return false }
     if r.IsInPeak(now) { return true }   // already in active window
     // Check if we're within ProvisionLeadS BEFORE next "up" transition
     nextT, kind := r.NextTransition(now)
     if kind != "up" { return false }
     lead := time.Duration(r.ProvisionLeadS) * time.Second
     return nextT.Sub(now) <= lead && nextT.Sub(now) > 0
