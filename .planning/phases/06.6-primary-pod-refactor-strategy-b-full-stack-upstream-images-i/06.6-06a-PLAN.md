// gateway/internal/primary/reconciler.go — added by THIS plan
func (r *Reconciler) Start(ctx context.Context) {
    // 1. Restart recovery (reviews #4) — always runs, regardless of DISABLED
    if err := r.recoverOpenLifecycle(ctx); err != nil {
        r.deps.Log.Error("primary recover-open-lifecycle failed", "err", err)
    }

    // 2. Event subscriber goroutine — ALWAYS launched, even if DISABLED (reviews #2)
    go r.runEventSubscriber(ctx)

    // 3. Schedule-tick dispatcher loop — gated by DISABLED at the tick decision point, NOT at Start()
    go r.runScheduleLoop(ctx)
}

func (r *Reconciler) runEventSubscriber(ctx context.Context) {
    pubsub := redisx.SubscribePrimaryEvents(ctx, r.deps.Redis)
    defer pubsub.Close()
    for msg := range pubsub.Channel() {
        var ev redisx.PrimaryEvent
        if err := json.Unmarshal([]byte(msg.Payload), &ev); err != nil {
            continue
        }
        if !r.isLeader.Load() {
            continue
        }
        switch ev.Type {
        case "force_up_request":
            r.handleForceUpRequest(ctx, ev)
        case "force_down_request":
            r.handleForceDownRequest(ctx, ev)
        }
    }
}

func (r *Reconciler) handleForceUpRequest(ctx context.Context, ev redisx.PrimaryEvent) {
    // Bypass schedule + cooldown — operator-explicit request
    state := r.deps.FSM.State()
    if state != StateAsleep {
        r.deps.Log.Info("primary force-up: not in Asleep; skipping", "state", state)
        return
    }
    r.deps.Log.Info("primary force-up: provisioning by operator request", "reason", ev.Reason)
    r.lastProvisionFailureAt.Store(nil)  // clear cooldown explicitly
    r.spawnProvisioning(ctx, "operator_force_up_"+ev.ReplicaID)
}

func (r *Reconciler) handleForceDownRequest(ctx context.Context, ev redisx.PrimaryEvent) {
    state := r.deps.FSM.State()
    if state != StateReady && state != StateDraining {
        r.deps.Log.Info("primary force-down: not in Ready/Draining; skipping", "state", state)
        return
    }
    r.deps.Log.Info("primary force-down: initiating drain by operator request", "reason", ev.Reason)
    r.startDrain(ctx, "operator_force_down_"+ev.ReplicaID)
}

func (r *Reconciler) runScheduleLoop(ctx context.Context) {
    // Leader election via redsync
    mu := redisx.NewPrimaryRedsync(r.deps.Redis).NewMutex(redisx.PrimaryLockKey(), redsync.WithExpiry(30*time.Second))
    ticker := time.NewTicker(1 * time.Second)
    defer ticker.Stop()
    for {
        select {
        case <-ctx.Done(): return
        case <-ticker.C:
            // Try to acquire/extend leadership
            if err := mu.LockContext(ctx); err != nil { r.isLeader.Store(false); continue }
            r.isLeader.Store(true)
            // SCHEDULE TICK GATING (reviews #2): DISABLED skips evaluateTick but Start() is already running event subscriber
            if r.cfg.PrimaryPodScheduleDisabled {
                // emit a heartbeat log every 60s for ops visibility but do NOT call evaluateTick
                continue
            }
            r.evaluateTick(ctx, time.Now(), r.deps.Log)
        }
    }
}

func (r *Reconciler) recoverOpenLifecycle(ctx context.Context) error {
    q := db.New(r.deps.DB)  // sqlc-generated bindings from Plan 06.6-07
    open, err := q.GetOpenPrimaryLifecycle(ctx)  // SELECT ... WHERE ended_at IS NULL LIMIT 1
    if err != nil {
        if errors.Is(err, pgx.ErrNoRows) { return nil }  // no open lifecycle, nothing to do
        return fmt.Errorf("query open primary lifecycle: %w", err)
    }
    if !open.VastInstanceID.Valid {
        // open row with no Vast instance ID — leftover from pre-provision crash, close
        _ = q.ClosePrimaryLifecycle(ctx, db.ClosePrimaryLifecycleParams{ID: open.ID, ShutdownReason: pgtype.Text{String: "gateway_restart_orphan_no_instance", Valid: true}})
        return nil
    }
    inst, err := r.deps.Vast.GetInstance(ctx, open.VastInstanceID.Int64)
    if err != nil || inst.ActualStatus != "running" {
        r.deps.Log.Warn("primary restart-recovery: instance not running, closing lifecycle as orphan", "lifecycle_id", open.ID, "instance_id", open.VastInstanceID.Int64, "err", err)
        _ = q.ClosePrimaryLifecycle(ctx, db.ClosePrimaryLifecycleParams{ID: open.ID, ShutdownReason: pgtype.Text{String: "gateway_restart_orphan", Valid: true}})
        return nil
    }
    // Reconstruct pod URLs from inst.Ports mapping (same logic as podLLMURL/podSTTURL/podEmbedURL/podDCGMURL helpers)
    urls := primaryPodURLs{
        LLM:   r.podLLMURL(inst),
        STT:   r.podSTTURL(inst),
        Embed: r.podEmbedURL(inst),
        DCGM:  r.podDCGMURL(inst),
    }
    // Health check ALL 4 — only restore if pod is fully healthy
    if !r.deps.HealthCheck(ctx, urls.LLM) || !r.deps.HealthCheck(ctx, urls.STT) || !r.deps.HealthCheck(ctx, urls.Embed) {
        r.deps.Log.Warn("primary restart-recovery: pod found but health check failed, closing as orphan", "lifecycle_id", open.ID)
        _ = q.ClosePrimaryLifecycle(ctx, db.ClosePrimaryLifecycleParams{ID: open.ID, ShutdownReason: pgtype.Text{String: "gateway_restart_orphan_unhealthy", Valid: true}})
        return nil
    }
    // Healthy! Restore in-memory state
    r.activeLifecycleID.Store(open.ID)
    r.activeInstanceID.Store(open.VastInstanceID.Int64)
    r.activePodURLs.Store(&urls)
    _ = r.deps.Loader.OverrideTier0("llm", urls.LLM)
    _ = r.deps.Loader.OverrideTier0("stt", urls.STT)
    _ = r.deps.Loader.OverrideTier0("embed", urls.Embed)
    r.deps.DCGMScraper.SetURL(urls.DCGM)
    r.deps.FSM.SetState(StateReady, time.Now(), "restart_recovery")
    r.deps.Log.Info("primary restart-recovery: rehydrated active lifecycle", "lifecycle_id", open.ID, "instance_id", open.VastInstanceID.Int64)
    return nil
}
