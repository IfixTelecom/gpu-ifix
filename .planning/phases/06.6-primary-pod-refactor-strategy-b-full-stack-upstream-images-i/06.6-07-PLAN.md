```sql
SET search_path = ai_gateway, public;

CREATE TABLE IF NOT EXISTS ai_gateway.primary_lifecycles (
    id                      BIGSERIAL PRIMARY KEY,
    started_at              TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    first_health_pass_at    TIMESTAMPTZ,
    drain_started_at        TIMESTAMPTZ,  -- NEW vs emerg (Phase 6.6)
    ended_at                TIMESTAMPTZ,
    trigger_reason          TEXT NOT NULL,
    vast_offer_id           BIGINT,
    vast_instance_id        BIGINT,
    accepted_dph            NUMERIC(6,4),
    total_cost_brl          NUMERIC(10,4),
    shutdown_reason         TEXT,
    events                  JSONB NOT NULL DEFAULT '[]'::JSONB,
    leader_replica          TEXT
);

CREATE UNIQUE INDEX IF NOT EXISTS primary_live_singleton
    ON ai_gateway.primary_lifecycles ((TRUE)) WHERE ended_at IS NULL;

CREATE INDEX IF NOT EXISTS idx_primary_lifecycles_started_at
    ON ai_gateway.primary_lifecycles (started_at DESC);
CREATE INDEX IF NOT EXISTS idx_primary_lifecycles_live
    ON ai_gateway.primary_lifecycles (ended_at) WHERE ended_at IS NULL;
CREATE INDEX IF NOT EXISTS idx_primary_lifecycles_month_cost
    ON ai_gateway.primary_lifecycles (started_at) WHERE ended_at IS NOT NULL;
-- +goose StatementEnd

-- +goose Down
-- +goose StatementBegin
DROP TABLE IF EXISTS ai_gateway.primary_lifecycles CASCADE;
-- +goose StatementEnd
```

<!-- 6 sqlc queries (5 standard + GetOpenPrimaryLifecycle NEW for reviews #4 restart recovery) -->

```sql
-- name: InsertPrimaryLifecycle :one
INSERT INTO ai_gateway.primary_lifecycles (trigger_reason, leader_replica)
VALUES ($1, $2)
RETURNING id, started_at;

-- name: UpdatePrimaryLifecycleVastIDs :exec
UPDATE ai_gateway.primary_lifecycles
SET vast_offer_id = $2, vast_instance_id = $3, accepted_dph = $4,
    events = events || $5::jsonb
WHERE id = $1;

-- name: MarkPrimaryLifecycleHealthy :exec
UPDATE ai_gateway.primary_lifecycles
SET first_health_pass_at = NOW(),
    events = events || $2::jsonb
WHERE id = $1;

-- name: MarkPrimaryLifecycleDraining :exec
UPDATE ai_gateway.primary_lifecycles
SET drain_started_at = NOW(),
    events = events || $2::jsonb
WHERE id = $1;

-- name: ClosePrimaryLifecycle :exec
UPDATE ai_gateway.primary_lifecycles
SET ended_at = NOW(),
    shutdown_reason = $2, total_cost_brl = $3,
    events = events || $4::jsonb
WHERE id = $1 AND ended_at IS NULL;

-- name: GetOpenPrimaryLifecycle :one
-- NEW per reviews consensus action #4 — restart recovery in primary.Reconciler.recoverOpenLifecycle
-- Returns the single open lifecycle row (ended_at IS NULL) or pgx.ErrNoRows.
-- Bounded LIMIT 1 because primary_live_singleton unique index guarantees at most 1.
SELECT id, started_at, first_health_pass_at, drain_started_at, ended_at,
       trigger_reason, vast_offer_id, vast_instance_id, accepted_dph,
       total_cost_brl, shutdown_reason, events, leader_replica
FROM ai_gateway.primary_lifecycles
WHERE ended_at IS NULL
LIMIT 1;
```
