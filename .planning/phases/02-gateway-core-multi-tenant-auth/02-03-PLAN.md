package auth

import "context"

type ctxKey int

const authCtxKey ctxKey = 0

// AuthContext is the payload placed on the request context when Verify
// succeeds. Downstream handlers (proxy, audit, idempotency) read it via
// FromContext. KeyPrefix is safe to log (last 4 chars only per D-A2).
type AuthContext struct {
	TenantID  string
	APIKeyID  string
	DataClass DataClass
	KeyPrefix string // "ifix_sk_****abcd"
}

// WithContext stashes ac on ctx.
func WithContext(ctx context.Context, ac AuthContext) context.Context {
	return context.WithValue(ctx, authCtxKey, ac)
}

// FromContext returns the AuthContext and true if present.
func FromContext(ctx context.Context) (AuthContext, bool) {
	ac, ok := ctx.Value(authCtxKey).(AuthContext)
	return ac, ok
}

// MustFromContext panics if no AuthContext is present. Use only in routes
// protected by auth.Middleware.
func MustFromContext(ctx context.Context) AuthContext {
	ac, ok := FromContext(ctx)
	if !ok {
		panic("auth: no AuthContext in ctx — middleware not applied")
	}
	return ac
}

package auth

import (
	"crypto/rand"
	"encoding/base32"
	"errors"
	"strings"

	"github.com/alexedwards/argon2id"
)

// KeyPrefix is the fixed prefix of every gateway API key. The prefix is
// deliberately distinctive so secret-scanning tools (GitHub, GitLab) and
// log redactors can spot leaks (CONTEXT.md D-A1).
const KeyPrefix = "ifix_sk_"

// keyBodyLen is the number of base32 chars after the prefix. 32 chars of
// rfc4648 base32 encode 20 random bytes = 160 bits of entropy, well above
// the 128-bit floor OWASP calls for on bearer tokens.
const keyBodyLen = 32

// keyTotalLen is len("ifix_sk_") + keyBodyLen = 40.
const keyTotalLen = len(KeyPrefix) + keyBodyLen

// DefaultParams are OWASP 2026 recommendations for argon2id bearer tokens.
// Tuned so verification on a 4 vCPU VPS takes ~30-50ms; cache TTL of 60s
// (Plan's cache.go) keeps that off the hot path.
var DefaultParams = &argon2id.Params{
	Memory:      64 * 1024, // 64 MiB
	Iterations:  3,
	Parallelism: 2,
	SaltLength:  16,
	KeyLength:   32,
}

// keyAlphabet is lowercase rfc4648 base32. Encoding uses NoPadding so the
// output length is deterministic. We lowercase post-encode because
// base32.StdEncoding produces uppercase by default.
var keyEncoding = base32.StdEncoding.WithPadding(base32.NoPadding)

// GenerateAPIKey produces a fresh (raw, argon2id-hash, prefix-for-display)
// triple. Call from the gatewayctl admin CLI; the raw value is returned
// to the operator ONCE and never stored in the database.
func GenerateAPIKey() (raw string, hash string, prefix string, err error) {
	b := make([]byte, 20) // 20 bytes = 32 base32 chars
	if _, err = rand.Read(b); err != nil {
		return "", "", "", err
	}
	raw = KeyPrefix + strings.ToLower(keyEncoding.EncodeToString(b))
	hash, err = argon2id.CreateHash(raw, DefaultParams)
	if err != nil {
		return "", "", "", err
	}
	prefix = KeyPrefix + "****" + raw[len(raw)-4:]
	return raw, hash, prefix, nil
}

// IsWellFormedKey returns true iff s syntactically matches ifix_sk_<32 lowercase base32>.
// Called from Verify BEFORE any DB query so malformed keys consume no
// resources. Uses constant-time character checks to avoid side-channels
// distinguishing "too short" from "bad char" from "bad prefix".
func IsWellFormedKey(s string) bool {
	if len(s) != keyTotalLen {
		return false
	}
	if !strings.HasPrefix(s, KeyPrefix) {
		return false
	}
	body := s[len(KeyPrefix):]
	for i := 0; i < len(body); i++ {
		c := body[i]
		// rfc4648 base32 lowercase: a-z (minus 0/1/8/9 not in alphabet) + 2-7
		if !((c >= 'a' && c <= 'z') || (c >= '2' && c <= '7')) {
			return false
		}
	}
	return true
}

// VerifyHash returns (true, nil) when raw matches hash; (false, nil) when
// it does not; and (false, err) on hash-parse errors.
func VerifyHash(raw, hash string) (bool, error) {
	match, err := argon2id.ComparePasswordAndHash(raw, hash)
	if err != nil {
		// Corrupt hash in DB — treat as mismatch after logging upstream.
		return false, err
	}
	return match, nil
}

// HashKey is a convenience wrapper used by tests that need a known
// argon2id hash for a known raw key.
func HashKey(raw string) (string, error) {
	return argon2id.CreateHash(raw, DefaultParams)
}

// Compile-time guard: refuse to ship if prefix length assumption changes.
var _ = errors.New // keep imports stable
```

package auth

import (
	"context"
	"crypto/sha256"
	"encoding/hex"
	"encoding/json"
	"errors"
	"time"

	"github.com/redis/go-redis/v9"
)

const (
	// cacheTTL is the positive/negative result cache duration. 60s balances
	// sub-ms hot path with revocation propagation (CONTEXT.md D-A2).
	cacheTTL = 60 * time.Second
)

// cacheEntry is the JSON payload stored in Redis. Includes Status so
// revoked keys are cached as such (avoids argon2-reverifying a known
// revoked key 10x/second during the 60s window).
type cacheEntry struct {
	TenantID  string    `json:"tenant_id"`
	APIKeyID  string    `json:"api_key_id"`
	DataClass DataClass `json:"data_class"`
	Status    string    `json:"status"` // "active" | "revoked"
	KeyPrefix string    `json:"key_prefix"`
}

func cacheKeyFor(rawKey string) string {
	sum := sha256.Sum256([]byte(rawKey))
	return "gw:apikey:" + hex.EncodeToString(sum[:])
}

func (v *Verifier) cacheGet(ctx context.Context, rawKey string) (cacheEntry, bool, error) {
	raw, err := v.redis.Get(ctx, cacheKeyFor(rawKey)).Bytes()
	if errors.Is(err, redis.Nil) {
		return cacheEntry{}, false, nil
	}
	if err != nil {
		return cacheEntry{}, false, err
	}
	var e cacheEntry
	if err := json.Unmarshal(raw, &e); err != nil {
		return cacheEntry{}, false, err
	}
	return e, true, nil
}

func (v *Verifier) cachePut(ctx context.Context, rawKey string, e cacheEntry) error {
	b, err := json.Marshal(e)
	if err != nil {
		return err
	}
	return v.redis.Set(ctx, cacheKeyFor(rawKey), b, cacheTTL).Err()
}
```

package auth

import (
	"context"
	"errors"
	"log/slog"
	"net/http"
	"strings"
	"time"

	"github.com/google/uuid"
	"github.com/jackc/pgx/v5/pgxpool"
	"github.com/redis/go-redis/v9"

	"github.com/ifixtelecom/gpu-ifix/gateway/internal/db/gen"
	"github.com/ifixtelecom/gpu-ifix/gateway/internal/httpx"
)

// Verifier resolves a raw API key to an AuthContext. DB misses are
// argon2id-verified; hits are cached in Redis for 60s (CONTEXT.md D-A2).
type Verifier struct {
	pool  *pgxpool.Pool
	q     *gen.Queries
	redis *redis.Client
	log   *slog.Logger
}

// NewVerifier wires the pool, queries, and redis client.
func NewVerifier(pool *pgxpool.Pool, redis *redis.Client, log *slog.Logger) *Verifier {
	return &Verifier{
		pool:  pool,
		q:     gen.New(pool),
		redis: redis,
		log:   log.With("module", "AUTH"),
	}
}

// ExtractKey returns the raw key per D-A5: Authorization: Bearer takes
// precedence over X-API-Key. Trims exactly "Bearer " (case-sensitive per
// RFC 6750 preferred practice).
func ExtractKey(r *http.Request) string {
	if h := r.Header.Get("Authorization"); h != "" {
		if strings.HasPrefix(h, "Bearer ") {
			return strings.TrimSpace(strings.TrimPrefix(h, "Bearer "))
		}
	}
	return strings.TrimSpace(r.Header.Get("X-API-Key"))
}

// Verify resolves rawKey to an AuthContext.
func (v *Verifier) Verify(ctx context.Context, rawKey string) (AuthContext, error) {
	if rawKey == "" {
		return AuthContext{}, ErrMissingAPIKey
	}
	if !IsWellFormedKey(rawKey) {
		return AuthContext{}, ErrMalformedKey
	}

	// 1. Cache fast path.
	if hit, found, err := v.cacheGet(ctx, rawKey); err == nil && found {
		return hitToAuth(hit)
	} else if err != nil {
		v.log.WarnContext(ctx, "auth cache get failed", "err", err)
		// Continue to DB; cache failures degrade perf but don't break auth.
	}

	// 2. Slow path: scan active keys, argon2-verify each hash.
	rows, err := v.q.ListActiveKeysAll(ctx)
	if err != nil {
		return AuthContext{}, errors.New("auth: db lookup failed")
	}
	for _, row := range rows {
		match, vErr := VerifyHash(rawKey, row.KeyHash)
		if vErr != nil {
			v.log.ErrorContext(ctx, "argon2 verify error", "err", vErr, "api_key_id", row.ID.String())
			continue
		}
		if !match {
			continue
		}
		// Matched. Cache positive result + touch last_used_at asynchronously.
		entry := cacheEntry{
			TenantID:  row.TenantID.String(),
			APIKeyID:  row.ID.String(),
			DataClass: DataClass(row.DataClass),
			Status:    string(row.Status),
			KeyPrefix: row.KeyPrefix,
		}
		_ = v.cachePut(ctx, rawKey, entry)
		// Best-effort touch; errors logged but don't block auth.
		go func(id uuid.UUID) {
			tctx, tcancel := context.WithTimeout(context.Background(), 3*time.Second)
			defer tcancel()
			if err := v.q.TouchKeyLastUsed(tctx, id); err != nil {
				v.log.Warn("touch last_used_at failed", "err", err, "api_key_id", id.String())
			}
		}(row.ID)
		return hitToAuth(entry)
	}

	// 3. No match: cache MISS briefly as "invalid" to avoid repeated
	//    argon2 passes on a brute-force attacker. Short TTL (5s) so real
	//    key issuance propagates quickly.
	_ = v.redis.Set(ctx, cacheKeyFor(rawKey)+":neg", "1", 5*time.Second).Err()
	return AuthContext{}, ErrInvalidAPIKey
}

func hitToAuth(e cacheEntry) (AuthContext, error) {
	switch e.Status {
	case "active":
		return AuthContext{
			TenantID:  e.TenantID,
			APIKeyID:  e.APIKeyID,
			DataClass: e.DataClass,
			KeyPrefix: e.KeyPrefix,
		}, nil
	case "revoked":
		return AuthContext{}, ErrRevokedAPIKey
	default:
		return AuthContext{}, ErrInvalidAPIKey
	}
}

// Middleware returns a chi-compatible middleware that enforces auth and
// threads AuthContext through the request. On failure, writes the OpenAI
// error envelope 401 (TEN-08 consistency).
func Middleware(v *Verifier, log *slog.Logger) func(http.Handler) http.Handler {
	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			rawKey := ExtractKey(r)
			ac, err := v.Verify(r.Context(), rawKey)
			if err != nil {
				status := http.StatusUnauthorized
				code := "invalid_api_key"
				msg := "The API key is invalid."
				switch {
				case errors.Is(err, ErrMissingAPIKey):
					code = "no_api_key"
					msg = "Missing API key. Provide Authorization: Bearer <key> or X-API-Key header."
				case errors.Is(err, ErrMalformedKey):
					code = "malformed_api_key"
					msg = "The API key is malformed. Expected format: ifix_sk_<32 chars>."
				case errors.Is(err, ErrRevokedAPIKey):
					code = "revoked_api_key"
					msg = "The API key has been revoked."
				}
				log.WarnContext(r.Context(), "auth rejected",
					"err", err.Error(),
					"request_id", httpx.RequestIDFrom(r.Context()),
					"key_prefix", safeKeyPrefix(rawKey),
				)
				httpx.WriteOpenAIError(w, status, "authentication_error", code, msg)
				return
			}
			ctx := WithContext(r.Context(), ac)
			// Enrich the request-scoped logger.
			if rl := httpx.LoggerFrom(ctx); rl != nil {
				ctx = httpx.WithLogger(ctx,
					rl.With("tenant_id", ac.TenantID, "api_key_id", ac.APIKeyID, "data_class", string(ac.DataClass)))
			}
			next.ServeHTTP(w, r.WithContext(ctx))
		})
	}
}

// safeKeyPrefix returns a log-safe preview of the raw key (first 8 + last 4).
// NEVER log the whole raw key; log only `ifix_sk_****abcd`.
func safeKeyPrefix(raw string) string {
	if len(raw) < 12 {
		return "ifix_sk_****"
	}
	return KeyPrefix + "****" + raw[len(raw)-4:]
}

## Trust Boundaries

| Boundary | Description |
|---|---|
| Client → gateway `/v1/*` | Untrusted Authorization/X-API-Key header; auth.Middleware is the trust gate |
| Operator terminal → gatewayctl | Operator is trusted; raw keys printed to stdout must not leak to other log sinks |
| Verifier → Postgres | Every Verify may read api_keys; cached in Redis for 60s |
| Verifier → Redis | Cache backing; failure degrades to DB but does not bypass auth |

## STRIDE Threat Register

| Threat ID | Category | Component | Severity | Disposition | Mitigation Plan |
|---|---|---|---|---|---|
| T-02-03-01 | Spoofing | API key brute force | high | mitigate | argon2id (64MiB/3/2) means ~30ms per guess on 4 vCPU → ~30 guesses/s max per core; effective entropy 160 bits; Redis 5s neg-cache on unknown keys further throttles rapid probes (gateway/internal/auth/apikey.go) |
| T-02-03-02 | Tampering | Timing attack comparing hashes | medium | mitigate | `argon2id.ComparePasswordAndHash` uses constant-time byte comparison internally; we never use `==` or `strings.Compare` on the raw key; short-circuit paths (Missing/Malformed) return BEFORE any per-row compare so the attacker can't distinguish "key unknown vs prefix wrong" via timing (gateway/internal/auth/apikey.go Verify) |
| T-02-03-03 | Elevation of Privilege | Header precedence confusion | high | mitigate | ExtractKey honors Authorization: Bearer over X-API-Key per D-A5; test `TestExtractKey_AuthorizationWins` asserts; precedence is deterministic (first match wins, no merging) (gateway/internal/auth/apikey.go) |
| T-02-03-04 | Information Disclosure | Raw key logged | high | mitigate | `safeKeyPrefix` returns only `ifix_sk_****<last4>`; gatewayctl key create prints raw to STDOUT but never through slog.Info with raw as an attribute; redactor still protects if a dev accidentally types `Authorization` as an attr name (gateway/internal/auth/apikey.go + gateway/cmd/gatewayctl/key.go) |
| T-02-03-05 | Information Disclosure | Raw key echoed in error message | medium | mitigate | Error messages contain no raw key — `err.Error()` bubbled through OpenAI envelope uses only sentinel labels (`no_api_key`, `invalid_api_key`, `revoked_api_key`, `malformed_api_key`) (gateway/internal/auth/apikey.go Middleware) |
| T-02-03-06 | Tampering | Cache poisoning | medium | mitigate | Redis key derived from `sha256(raw_key)`; attacker must know the raw key to poison its cache entry (at which point they already have it). Cache stores signed-by-origin data (tenant_id pulled from authoritative DB row, not from request) (gateway/internal/auth/cache.go) |
| T-02-03-07 | DoS | Argon2 CPU exhaustion via invalid keys | high | mitigate | IsWellFormedKey short-circuits before any argon2; malformed key → constant-time reject. Neg-cache on unknown keys (5s) prevents repeated argon2 sweeps (gateway/internal/auth/apikey.go) |
| T-02-03-08 | Repudiation | Admin actions unauditable | medium | mitigate | gatewayctl tenant/key actions log via slog with key_prefix + tenant_id; production deploys capture gatewayctl stdout + stderr to persistent logs via Portainer stack config (Plan 02-08) |
| T-02-03-09 | Information Disclosure | Raw key stored in DB | high | mitigate | Only argon2id hash + prefix are persisted; raw key never crosses gateway → DB boundary. Verified: no `key=` column in api_keys, only `key_hash` and `key_prefix` (gateway/db/migrations/0002_create_api_keys.sql) |
| T-02-03-10 | Elevation of Privilege | SQL injection via tenant slug | high | mitigate | All writes use sqlc parameterized queries (`$1`, `$2`). No string concatenation anywhere in gatewayctl tenant/key. Verified in gateway/db/queries/admin.sql |
