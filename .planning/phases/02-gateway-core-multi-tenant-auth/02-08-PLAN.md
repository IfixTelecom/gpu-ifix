FILE `gateway/Dockerfile`:

```dockerfile
# syntax=docker/dockerfile:1.7

# =========================================================================
# Stage 1: builder — Go toolchain + gateway source + sqlc-generated code
# =========================================================================
FROM golang:1.23-alpine AS builder
WORKDIR /build

RUN apk add --no-cache ca-certificates git

# Install sqlc so `go generate` (if we added one) could regenerate.
# For Phase 2 we commit the sqlc output, so this RUN is a safety net —
# if CI forgot to run `sqlc generate` before build, the resulting binary
# still compiles against a stale gen/ directory.
#
# Note: the CI job runs sqlc generate BEFORE `docker build`; this stage
# only verifies the generated code compiles.

# Dependency layer cache: copy go.mod/go.sum first so source changes
# don't bust the module-download cache.
COPY go.mod go.sum ./
RUN go mod download

# Copy shared pkg/openai and the entire gateway subtree.
COPY pkg/openai ./pkg/openai
COPY gateway ./gateway

# Build version metadata injected via -ldflags. CI sets GATEWAY_VERSION;
# local builds fall back to "dev".
ARG GATEWAY_VERSION=dev

RUN CGO_ENABLED=0 GOOS=linux GOARCH=amd64 \
    go build -trimpath \
    -ldflags="-s -w -X github.com/ifixtelecom/gpu-ifix/gateway/internal/obs.BuildVersion=${GATEWAY_VERSION}" \
    -o /gateway \
    ./gateway/cmd/gateway

RUN CGO_ENABLED=0 GOOS=linux GOARCH=amd64 \
    go build -trimpath \
    -ldflags="-s -w -X github.com/ifixtelecom/gpu-ifix/gateway/internal/obs.BuildVersion=${GATEWAY_VERSION}" \
    -o /gatewayctl \
    ./gateway/cmd/gatewayctl

# =========================================================================
# Stage 2: runtime — distroless static (no shell, no libc, ~2 MB base)
# =========================================================================
FROM gcr.io/distroless/static-debian12 AS runtime

# Timezone database so TZ=America/Sao_Paulo resolves at runtime.
COPY --from=builder /usr/share/zoneinfo /usr/share/zoneinfo
ENV TZ=America/Sao_Paulo

COPY --from=builder /gateway /gateway
COPY --from=builder /gatewayctl /gatewayctl

# Gateway HTTP port (fixed per CONTEXT.md Plumbing: :8080 on VPS; Traefik
# / DNS added in Fase 10).
EXPOSE 8080

# SIGTERM is the default signal docker sends on `docker stop`; match it
# explicitly so graceful shutdown (25s budget) kicks in reliably.
STOPSIGNAL SIGTERM

# Default entrypoint is the gateway server. To run gatewayctl:
#   docker run --rm --entrypoint /gatewayctl <image> migrate up
ENTRYPOINT ["/gateway"]
```

```yaml
name: build-gateway

on:
  push:
    branches: [main, develop]
    tags: ['v*']
  pull_request:
    branches: [main, develop]
    paths:
      - 'gateway/**'
      - 'pkg/openai/**'
      - 'go.mod'
      - 'go.sum'
      - '.github/workflows/build-gateway.yml'
  workflow_dispatch:
    inputs:
      tag:
        description: 'Optional explicit tag (else branch-sha)'
        required: false
        type: string

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

permissions:
  contents: read
  packages: write

env:
  REGISTRY: ghcr.io
  IMAGE_GATEWAY: ghcr.io/ifixtelecom/ifix-ai-gateway
  GO_VERSION: '1.23'
  SQLC_VERSION: 'v1.27.0'

jobs:
  test:
    name: Go unit tests + sqlc codegen verify
    runs-on: ubuntu-latest
    timeout-minutes: 10
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Go
        uses: actions/setup-go@v5
        with:
          go-version: ${{ env.GO_VERSION }}
          cache: true

      - name: Install sqlc
        run: |
          go install github.com/sqlc-dev/sqlc/cmd/sqlc@${{ env.SQLC_VERSION }}

      - name: Generate sqlc code (gateway/)
        working-directory: gateway
        run: |
          sqlc generate
          # Fail build if sqlc output would change the tree — force every
          # PR/push to commit regenerated files alongside query changes.
          if ! git diff --exit-code internal/db/gen/; then
            echo "::error::sqlc-generated code is out of sync with queries/. Run \`cd gateway && sqlc generate\` and commit the output."
            exit 1
          fi

      - name: go vet
        run: go vet ./gateway/... ./pkg/openai/... ./pod/...

      - name: go test (unit only — no integration tag)
        run: go test ./gateway/... ./pkg/openai/... ./pod/... -count=1 -race -timeout=5m

      - name: gofmt check
        run: |
          UNFORMATTED="$(gofmt -l .)"
          if [[ -n "${UNFORMATTED}" ]]; then
            echo "::error::gofmt unclean files:"
            echo "${UNFORMATTED}"
            exit 1
          fi

  integration-test:
    name: Integration tests (testcontainers-go)
    runs-on: ubuntu-latest
    needs: [test]
    timeout-minutes: 15
    # GitHub-hosted Ubuntu runners have docker available by default.
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Go
        uses: actions/setup-go@v5
        with:
          go-version: ${{ env.GO_VERSION }}
          cache: true

      - name: Install sqlc
        run: go install github.com/sqlc-dev/sqlc/cmd/sqlc@${{ env.SQLC_VERSION }}

      - name: Generate sqlc code
        working-directory: gateway
        run: sqlc generate

      - name: Integration tests
        run: |
          go test -tags=integration ./gateway/internal/integration_test/... -count=1 -v -timeout=10m

  compute-tags:
    name: Compute image tags
    runs-on: ubuntu-latest
    needs: [test]
    outputs:
      gateway_tags:      ${{ steps.set.outputs.gateway_tags }}
      is_stable_release: ${{ steps.set.outputs.is_stable_release }}
      gateway_version:   ${{ steps.set.outputs.gateway_version }}
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Compute tags (mirrors build-pod.yml)
        id: set
        env:
          REF: ${{ github.ref }}
          SHA: ${{ github.sha }}
          INPUT_TAG: ${{ inputs.tag }}
        run: |
          set -euo pipefail
          SHORT_SHA="$(echo "${SHA}" | cut -c1-7)"
          GW_TAGS=""
          IS_STABLE="false"
          GW_VERSION=""

          if [[ "${REF}" == refs/tags/v* ]]; then
            VER="${REF##refs/tags/}"
            GW_TAGS="${IMAGE_GATEWAY}:${VER}
          ${IMAGE_GATEWAY}:latest
          ${IMAGE_GATEWAY}:${VER}-${SHORT_SHA}"
            IS_STABLE="true"
            GW_VERSION="${VER}"
          elif [[ "${REF}" == refs/heads/main ]]; then
            GW_TAGS="${IMAGE_GATEWAY}:main
          ${IMAGE_GATEWAY}:main-${SHORT_SHA}"
            GW_VERSION="main-${SHORT_SHA}"
          elif [[ "${REF}" == refs/heads/develop ]]; then
            GW_TAGS="${IMAGE_GATEWAY}:develop
          ${IMAGE_GATEWAY}:develop-${SHORT_SHA}
          ${IMAGE_GATEWAY}:latest-dev"
            GW_VERSION="develop-${SHORT_SHA}"
          elif [[ -n "${INPUT_TAG}" ]]; then
            GW_TAGS="${IMAGE_GATEWAY}:${INPUT_TAG}"
            GW_VERSION="${INPUT_TAG}"
          else
            GW_TAGS="${IMAGE_GATEWAY}:pr-${SHORT_SHA}"
            GW_VERSION="pr-${SHORT_SHA}"
          fi

          {
            echo "gateway_tags<<EOF"
            printf '%s\n' "${GW_TAGS}"
            echo "EOF"
            printf 'is_stable_release=%s\n' "${IS_STABLE}"
            printf 'gateway_version=%s\n' "${GW_VERSION}"
          } >> "$GITHUB_OUTPUT"

          echo "resolved tags:"; echo "${GW_TAGS}"
          echo "version: ${GW_VERSION}"

  build-gateway:
    name: Build & push gateway image
    runs-on: ubuntu-latest
    needs: [test, integration-test, compute-tags]
    timeout-minutes: 20
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Go (for sqlc generate)
        uses: actions/setup-go@v5
        with:
          go-version: ${{ env.GO_VERSION }}
          cache: true

      - name: Install sqlc
        run: go install github.com/sqlc-dev/sqlc/cmd/sqlc@${{ env.SQLC_VERSION }}

## Trust Boundaries

| GitHub Actions → GHCR | OIDC-backed push via built-in GITHUB_TOKEN; no long-lived creds |

| Portainer stack → container | Env vars injected with secrets (PG DSN, Redis password, Sentry DSN) |
| Image (distroless) → runtime | No shell, no libc, no package manager — minimal attack surface |

| Threat ID | Category | Component | Severity | Disposition | Mitigation Plan |
|---|---|---|---|---|---|
| T-02-08-01 | Tampering | Image supply chain | high | mitigate | Two-stage build with distroless runtime (no shell, no package manager, no utilities to exploit if an attacker gets RCE). Build uses ONLY pinned golang:1.23-alpine base. `-trimpath` strips build paths so binaries aren't environment-correlated (gateway/Dockerfile) |
| T-02-08-02 | Elevation of Privilege | Root user in container | medium | mitigate | Distroless/static-debian12 runs as `nonroot` user by default (uid 65532); our Dockerfile inherits that — no explicit USER instruction needed (distroless default is secure) |
| T-02-08-03 | Information Disclosure | .env file baked into image | high | mitigate | `.dockerignore` excludes `.env*`; docker-compose.yml uses `${VAR}` interpolation only; no secrets in git (gateway/.dockerignore) |
| T-02-08-04 | Tampering | Stale sqlc-generated code vs migrations | high | mitigate | CI `test` job runs `sqlc generate` + `git diff --exit-code internal/db/gen/` — any drift fails the build with a clear error message (.github/workflows/build-gateway.yml) |
| T-02-08-05 | DoS | Migrations on every boot lock schema | medium | mitigate | `AI_GATEWAY_MIGRATE_ON_BOOT=true` documented as flip-to-false after first deploy; goose marks applied versions so subsequent Up is idempotent (gateway/docker-compose.yml) |
