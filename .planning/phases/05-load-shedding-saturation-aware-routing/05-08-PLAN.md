  //go:build integration

  package integration_test

  import (
      "context"
      "fmt"
      "net/http"
      "net/http/httptest"
      "sync/atomic"
      "testing"
      "time"

      "github.com/jackc/pgx/v5/pgxpool"
      "github.com/redis/go-redis/v9"
  )

  // ShedStack is the bundle of primitives needed by SC-1..SC-4 tests.
  type ShedStack struct {
      T         *testing.T
      Ctx       context.Context
      Pool      *pgxpool.Pool
      Rdb       *redis.Client

      // Mock upstreams exposed for test control
      Tier0Mock *ControlledMockServer // local-llm target
      Tier1Mock *ControlledMockServer // openrouter-chat target

      // GatewayURL is the URL of the gateway under test (in-process httptest server)
      GatewayURL string

      // Cleanup functions registered via t.Cleanup — caller does not need to call manually
  }

  // ControlledMockServer simulates an upstream with controllable latency + status.
  type ControlledMockServer struct {
      srv      *httptest.Server
      latency  atomic.Int64 // ns; read by handler per request
      statusCode atomic.Int32 // default 200; set to 503/429 to simulate failure
      hits     atomic.Int64
  }

  func newControlledMock(t *testing.T) *ControlledMockServer {
      t.Helper()
      m := &ControlledMockServer{}
      m.statusCode.Store(200)
      m.srv = httptest.NewServer(http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
          m.hits.Add(1)
          if lat := m.latency.Load(); lat > 0 {
              time.Sleep(time.Duration(lat))
          }
          code := m.statusCode.Load()
          w.WriteHeader(int(code))
          if code < 400 {
              // OpenAI-compatible response skeleton
              fmt.Fprintf(w, `{"id":"mock-%d","choices":[{"message":{"role":"assistant","content":"ok"}}],"usage":{"prompt_tokens":10,"completion_tokens":5,"total_tokens":15}}`, m.hits.Load())
          } else {
              fmt.Fprintf(w, `{"error":{"message":"mock-%d","type":"mock","code":"mock_error"}}`, code)
          }
      }))
      t.Cleanup(m.srv.Close)
      return m
  }

  func (m *ControlledMockServer) URL() string                { return m.srv.URL }
  func (m *ControlledMockServer) SetLatency(d time.Duration) { m.latency.Store(d.Nanoseconds()) }
  func (m *ControlledMockServer) SetStatus(code int)         { m.statusCode.Store(int32(code)) }
  func (m *ControlledMockServer) Hits() int64                { return m.hits.Load() }
  func (m *ControlledMockServer) ResetHits()                 { m.hits.Store(0) }

  // newShedStack boots Postgres + Redis + 2 mock upstreams, seeds DB with
  // migrations 0001..0017, and returns a ShedStack ready for SC tests.
  // The gateway itself is NOT booted here — tests boot via helper
  // bootGateway(stack, env overrides...).
  func newShedStack(t *testing.T) *ShedStack {
      t.Helper()
      ctx, cancel := context.WithTimeout(context.Background(), 120*time.Second)
      t.Cleanup(cancel)

      pool, rdb := freshSchema(t, ctx)

      t0 := newControlledMock(t)
      t1 := newControlledMock(t)

      // Seed upstreams: local-llm → t0.URL, openrouter-chat → t1.URL
      _, err := pool.Exec(ctx, `
          INSERT INTO ai_gateway.upstreams (name, role, tier, enabled, endpoint, circuit_config)
          VALUES
              ('local-llm','llm',0,true, $1, '{"failures":3,"cooldown_s":30,"shed_inflight_max":4,"shed_p95_ms":500,"shed_vram_used_mib":21504,"shed_arm_seconds":1,"shed_recover_seconds":2}'::jsonb),
              ('openrouter-chat','llm',1,true, $2, '{"failures":3,"cooldown_s":30}'::jsonb)
          ON CONFLICT (name) DO UPDATE SET endpoint=EXCLUDED.endpoint, circuit_config=EXCLUDED.circuit_config
      `, t0.URL(), t1.URL())
      if err != nil {
          t.Fatalf("seed upstreams: %v", err)
      }

      // Short hysteresis for tests (arm=1s, recover=2s) so tests run fast.

      return &ShedStack{
          T:          t,
          Ctx:        ctx,
          Pool:       pool,
          Rdb:        rdb,
          Tier0Mock:  t0,
          Tier1Mock:  t1,
          GatewayURL: "", // set by bootGateway
      }
  }

  // bootGateway starts the gateway in-process using the test Postgres+Redis.
  // Returns teardown via t.Cleanup. GatewayURL is populated on stack.
  //
  // NOTE: The exact signature depends on how cmd/gateway/main.go exposes a
  // testable entry point. If main.go has a func Run(ctx, cfg) error pattern
  // (refactored by Plan 06), call it directly. Otherwise, this helper may
  // need to import the main package's router builder (buildRouter) and serve
  // it via httptest.NewServer. Plan 06 should have provided the entry hook.
  func bootGateway(stack *ShedStack, envOverrides map[string]string) string {
      // IMPLEMENTATION NOTE (to be filled in execute):
      //
      // Option A: If cmd/gateway/main.go exposes a buildRouter or Run(ctx)
      // function that accepts custom pool + rdb + mock clock, use that directly.
      //
      // Option B: If main.go only runs via flags/env, use os.Setenv to override
      // GATEWAY_DB_URL + GATEWAY_REDIS_ADDR + DCGM_EXPORTER_URL=""
      // (disable DCGM for tests — VRAM signal reduces to unknown, FSM uses 1-of-2).
      // Then build the router inline using the exported package API.
      //
      // A simpler option: since shed Middleware + Set + Ticker are testable
      // in isolation (Plan 06), skip full gateway boot for SC-1..SC-4 and
      // test at the middleware layer with a minimal chi.Router that wires
      // shed.Middleware + audit.Middleware + a simple upstream proxy.
      //
      // The chosen approach is documented here for subsequent tasks to follow.
      return "http://127.0.0.1:0/PLACEHOLDER"
  }

  // sqlUpdate helper for hot-reload tests
  func sqlUpdate(t *testing.T, stack *ShedStack, q string, args ...any) {
      t.Helper()
      if _, err := stack.Pool.Exec(stack.Ctx, q, args...); err != nil {
          t.Fatalf("sql: %v", err)
      }
  }

  //go:build integration

  package integration_test

  import (
      "net/http"
      "testing"
      "time"

      vegeta "github.com/tsenart/vegeta/lib"
  )

  // TestSC1_BurstExceedsTenantCapOverflowsToTier1 validates Success Criterion 1:
  //   "Under a burst where inflight on local LLM exceeds the configured slot
  //    count, excess requests are routed to OpenRouter automatically; below
  //    threshold, traffic returns to local."
  //
  // Scenario:
  //   - Tenant cap_llm = 4 (seeded via migration 0016)
  //   - Tier-0 mock has latency 300ms (to keep inflight high at 50 RPS)
  //   - Attack 50 RPS for 20s = ~1000 requests
  //   - Expect: many requests served by tier-1 mock (overflow)
  //   - After attack: FSM should return to StateOff within ~90s (arm=1s + recover=2s in test config)
  func TestSC1_BurstExceedsTenantCapOverflowsToTier1(t *testing.T) {
      stack := newShedStack(t)
      gwURL := bootGateway(stack, nil)

      // Slow upstream → inflight builds up
      stack.Tier0Mock.SetLatency(300 * time.Millisecond)
      stack.Tier0Mock.ResetHits()
      stack.Tier1Mock.ResetHits()

      target := vegeta.Target{
          Method: "POST",
          URL:    gwURL + "/v1/chat/completions",
          Header: http.Header{
              "Authorization": {"Bearer TEST_CONVERSEAI_KEY"}, // seeded in freshSchema
              "Content-Type":  {"application/json"},
          },
          Body: []byte(`{"model":"qwen","messages":[{"role":"user","content":"hi"}]}`),
      }
      targeter := vegeta.NewStaticTargeter(target)
      rate := vegeta.Rate{Freq: 50, Per: time.Second}
      attacker := vegeta.NewAttacker()

      var metrics vegeta.Metrics
      for res := range attacker.Attack(targeter, rate, 20*time.Second, "SC-1 burst") {
          metrics.Add(res)
      }
      metrics.Close()

      tier0Hits := stack.Tier0Mock.Hits()
      tier1Hits := stack.Tier1Mock.Hits()
      t.Logf("SC-1 results: tier-0 hits=%d, tier-1 hits=%d, success=%f, p99=%s",
          tier0Hits, tier1Hits, metrics.Success, metrics.Latencies.P99)

      // Assert overflow happened
      if tier1Hits < 50 {
          t.Errorf("expected ≥50 tier-1 hits (overflow), got %d", tier1Hits)
      }
      // Success rate should be high (tier-1 serves 2xx)
      if metrics.Success < 0.95 {
          t.Errorf("success rate %f below 0.95", metrics.Success)
      }

      // Recovery phase: stop latency, wait for FSM to return to OFF
      stack.Tier0Mock.SetLatency(0)
      // Allow ~90s for FSM to go through RECOVERING with test-scaled arm=1/recover=2 seconds
      // In practice FSM needs ~3 seconds but give budget for test scheduler flakiness
      deadline := time.Now().Add(90 * time.Second)
      for time.Now().Before(deadline) {
          // Poll shed state via Redis directly (simpler than gateway /admin endpoint)
          m, _ := readShedState(stack, "local-llm")
          if m["state"] == "off" {
              return // SC-1 PASS
          }
          time.Sleep(500 * time.Millisecond)
      }
      t.Errorf("FSM did not return to off within 90s after burst ended")
  }

  // readShedState reads the Redis mirror Hash for an upstream.
  func readShedState(stack *ShedStack, upstream string) (map[string]string, error) {
      return stack.Rdb.HGetAll(stack.Ctx, "gw:shed:"+upstream).Result()
  }

  //go:build integration

  package integration_test

  import (
      "testing"
      "time"
  )

  // TestSC3_HotReloadAppliesInUnder2Seconds validates Success Criterion 3:
  //   "Thresholds (inflight_max, P95_ms, VRAM_bytes, hysteresis_seconds)
  //    can be changed by updating rows in Postgres and take effect within 2s
  //    without restarting the gateway."
  //
  // Scenario:
  //   - Drive FSM into StateOn via slow tier-0 + many concurrent requests
  //   - UPDATE circuit_config.shed_inflight_max to very high (disable shed)
  //   - Assert FSM transitions to Recovering within 2s of NOTIFY upstreams_changed
  //   - FSM reaches Off after recover_seconds (preserves hysteresis — D-C5)
  func TestSC3_HotReloadAppliesInUnder2Seconds(t *testing.T) {
      stack := newShedStack(t)
      _ = bootGateway(stack, nil)

      // Drive FSM to ON via latency spike + concurrent inflight burst.
      stack.Tier0Mock.SetLatency(600 * time.Millisecond) // P95 > 500ms threshold
      // Simulate sustained inflight — ideally launching 10 concurrent clients.
      // For integration_test simplicity, we bypass HTTP and directly write to
      // the Inflight registry via a test hook, OR drive via vegeta for 5s.
      //
      // Here we prefer the vegeta approach for realism.
      hammer := func(d time.Duration) {
          target := buildChatTarget(stack)
          targeter := vegeta.NewStaticTargeter(target)
          rate := vegeta.Rate{Freq: 10, Per: time.Second}
          attacker := vegeta.NewAttacker()
          for range attacker.Attack(targeter, rate, d, "sc3 warmup") { }
      }
      hammer(5 * time.Second)

      // Confirm FSM=ON
      stateBefore := waitForState(t, stack, "local-llm", "on", 10*time.Second)
      if stateBefore != "on" {
          t.Fatalf("precondition FSM=on not reached within 10s; got %s", stateBefore)
      }

      // Hot-reload: raise inflight threshold to 1000 (effectively disable shed)
      updateStart := time.Now()
      sqlUpdate(t, stack, `
          UPDATE ai_gateway.upstreams
          SET circuit_config = circuit_config || jsonb_build_object('shed_inflight_max', 1000)
          WHERE name = 'local-llm'
      `)

      // Assert FSM reacts within 2s — transitions ON→RECOVERING (hysteresis preserved)
      reachedRecovering := false
      deadline := time.Now().Add(2 * time.Second)
      for time.Now().Before(deadline) {
          m, _ := readShedState(stack, "local-llm")
          if m["state"] == "recovering" || m["state"] == "off" {
              reachedRecovering = true
              break
          }
          time.Sleep(100 * time.Millisecond)
      }
      elapsed := time.Since(updateStart)
      if !reachedRecovering {
          t.Errorf("FSM did not transition out of ON within 2s of UPDATE; elapsed=%s", elapsed)
      }
      t.Logf("SC-3 PASS: hot-reload took %s", elapsed)
  }

  // buildChatTarget builds a vegeta target for the /v1/chat/completions route
  // using the seeded ConverseAI API key.
  func buildChatTarget(stack *ShedStack) vegeta.Target {
      return vegeta.Target{
          Method: "POST",
          URL:    stack.GatewayURL + "/v1/chat/completions",
          Body:   []byte(`{"model":"qwen","messages":[{"role":"user","content":"hi"}]}`),
      }
  }

  // waitForState polls Redis mirror until state == target or timeout expires.
  func waitForState(t *testing.T, stack *ShedStack, upstream, target string, timeout time.Duration) string {
      t.Helper()
      deadline := time.Now().Add(timeout)
      var last string
      for time.Now().Before(deadline) {
          m, _ := readShedState(stack, upstream)
          last = m["state"]
          if last == target {
              return target
          }
          time.Sleep(100 * time.Millisecond)
      }
      return last
