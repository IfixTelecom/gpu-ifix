  ```go
  package main

  import (
      "context"
      "encoding/json"
      "flag"
      "fmt"
      "log/slog"
      "os"
      "strings"
      "text/tabwriter"
      "time"

      "github.com/ifixtelecom/gpu-ifix/gateway/internal/redisx"
  )

  // runShed is the top-level dispatcher for the `shed-state` and `shed-force`
  // subcommands. Called from main.go when os.Args[1] == "shed-state" or
  // "shed-force".
  func runShedState(ctx context.Context, args []string, log *slog.Logger) int {
      fs := flag.NewFlagSet("shed-state", flag.ContinueOnError)
      upstream := fs.String("upstream", "", "restrict to a single upstream")
      format := fs.String("format", "table", "output format: table | json")
      if err := fs.Parse(args); err != nil {
          return 2
      }

      rdb := mustRedisClient(log)
      defer rdb.Close()

      type row struct {
          Upstream       string `json:"upstream"`
          State          string `json:"state"`
          SinceUnix      string `json:"since_unix"`
          Reason         string `json:"reason"`
          ForceActive    bool   `json:"force_active"`
          ForceState     string `json:"force_state,omitempty"`
          ForceTTLSeconds int64 `json:"force_ttl_seconds,omitempty"`
      }

      var rows []row

      keys, err := redisx.AllShedStateKeys(ctx, rdb)
      if err != nil {
          fmt.Fprintf(os.Stderr, "list shed keys: %v\n", err)
          return 1
      }

      for _, k := range keys {
          // Strip "gw:shed:" prefix to get upstream name
          name := strings.TrimPrefix(k, "gw:shed:")
          if *upstream != "" && name != *upstream {
              continue
          }
          m, _ := redisx.ReadShedState(ctx, rdb, name)
          forceState, forceTTL, hasForce := redisx.GetShedForce(ctx, rdb, name)
          r := row{
              Upstream:   name,
              State:      m["state"],
              SinceUnix:  m["since_unix"],
              Reason:     m["reason"],
              ForceActive: hasForce,
              ForceState:  forceState,
          }
          if hasForce {
              r.ForceTTLSeconds = int64(forceTTL / time.Second)
          }
          rows = append(rows, r)
      }

      switch *format {
      case "json":
          out, _ := json.MarshalIndent(rows, "", "  ")
          fmt.Println(string(out))
      case "table":
          tw := tabwriter.NewWriter(os.Stdout, 0, 2, 2, ' ', 0)
          fmt.Fprintln(tw, "UPSTREAM\tSTATE\tSINCE_UNIX\tREASON\tFORCE\tTTL_S")
          for _, r := range rows {
              force := "-"
              if r.ForceActive {
                  force = r.ForceState
              }
              fmt.Fprintf(tw, "%s\t%s\t%s\t%s\t%s\t%d\n",
                  r.Upstream, r.State, r.SinceUnix, r.Reason, force, r.ForceTTLSeconds)
          }
          _ = tw.Flush()
      default:
          fmt.Fprintf(os.Stderr, "unknown format %q (want table|json)\n", *format)
          return 2
      }

      return 0
  }

  func runShedForce(ctx context.Context, args []string, log *slog.Logger) int {
      if len(args) < 1 {
          fmt.Fprintln(os.Stderr, "Usage: gatewayctl shed-force {on|off|clear} --upstream X [--ttl 300s]")
          return 2
      }
      action := args[0]
      fs := flag.NewFlagSet("shed-force", flag.ContinueOnError)
      upstream := fs.String("upstream", "", "target upstream (required)")
      ttlStr := fs.String("ttl", "300s", "TTL for on/off (default 300s, max 1h)")
      if err := fs.Parse(args[1:]); err != nil {
          return 2
      }
      if *upstream == "" {
          fmt.Fprintln(os.Stderr, "--upstream required")
          return 2
      }

      rdb := mustRedisClient(log)
      defer rdb.Close()

      switch action {
      case "on", "off":
          ttl, err := time.ParseDuration(*ttlStr)
          if err != nil {
              fmt.Fprintf(os.Stderr, "invalid ttl %q: %v\n", *ttlStr, err)
              return 2
          }
          if err := redisx.WriteShedForce(ctx, rdb, *upstream, action, ttl); err != nil {
              fmt.Fprintf(os.Stderr, "shed-force: %v\n", err)
              return 1
          }
          fmt.Fprintf(os.Stdout, "shed-force: %s=%s (ttl %s)\n", *upstream, action, ttl)
          return 0
      case "clear":
          if err := redisx.DeleteShedForce(ctx, rdb, *upstream); err != nil {
              fmt.Fprintf(os.Stderr, "shed-force clear: %v\n", err)
              return 1
          }
          fmt.Fprintf(os.Stdout, "shed-force: %s cleared\n", *upstream)
          return 0
      default:
          fmt.Fprintf(os.Stderr, "unknown action %q (want on|off|clear)\n", action)
          return 2
      }
  }
  ```

  ```go
  // runThresholds is the dispatcher for `thresholds` subcommand family.
  func runThresholds(ctx context.Context, args []string, log *slog.Logger) int {
      if len(args) == 0 {
          fmt.Fprintln(os.Stderr, "Usage: gatewayctl thresholds set [flags]")
          return 2
      }
      switch args[0] {
      case "set":
          return runThresholdsSet(ctx, args[1:], log)
      default:
          fmt.Fprintf(os.Stderr, "unknown thresholds subcommand: %s\n", args[0])
          return 2
      }
  }

  func runThresholdsSet(ctx context.Context, args []string, log *slog.Logger) int {
      fs := flag.NewFlagSet("thresholds set", flag.ContinueOnError)
      upstream := fs.String("upstream", "", "upstream name (required)")
      inflight := fs.Int("inflight", -1, "shed_inflight_max (>=1); -1 = unchanged")
      p95 := fs.Int("p95-ms", -1, "shed_p95_ms (>=1); -1 = unchanged")
      vramMiB := fs.Int64("vram-mib", -1, "shed_vram_used_mib (>=1); -1 = unchanged. NOTE: MiB not bytes")
      armS := fs.Int("arm-s", -1, "shed_arm_seconds (>=5); -1 = unchanged")
      recoverS := fs.Int("recover-s", -1, "shed_recover_seconds (>=5); -1 = unchanged")
      if err := fs.Parse(args); err != nil {
          return 2
      }
      if *upstream == "" {
          fmt.Fprintln(os.Stderr, "--upstream required")
          return 2
      }

      pool := mustPool(log)
      defer pool.Close()
      q := gen.New(pool)

      // Fetch current upstream row to merge JSONB
      row, err := q.GetUpstreamByName(ctx, *upstream)
      if err != nil {
          fmt.Fprintf(os.Stderr, "upstream %q not found: %v\n", *upstream, err)
          return 1
      }

      merged := map[string]any{}
      if len(row.CircuitConfig) > 0 {
          _ = json.Unmarshal(row.CircuitConfig, &merged)
      }

      anyFlag := false
      if *inflight >= 0 {
          if *inflight == 0 {
              fmt.Fprintln(os.Stderr, "--inflight must be >= 1 (use 0 only to disable via manual SQL)")
              return 2
          }
          merged["shed_inflight_max"] = *inflight
          anyFlag = true
      }
      if *p95 >= 0 {
          if *p95 == 0 {
              fmt.Fprintln(os.Stderr, "--p95-ms must be >= 1")
              return 2
          }
          merged["shed_p95_ms"] = *p95
          anyFlag = true
      }
      if *vramMiB >= 0 {
          if *vramMiB == 0 || *vramMiB > 1_000_000 {
              fmt.Fprintln(os.Stderr, "--vram-mib must be in [1, 1_000_000]")
              return 2
          }
          merged["shed_vram_used_mib"] = *vramMiB
          anyFlag = true
      }
      if *armS >= 0 {
          if *armS < 5 {
              fmt.Fprintln(os.Stderr, "--arm-s must be >= 5")
              return 2
          }
          merged["shed_arm_seconds"] = *armS
          anyFlag = true
      }
      if *recoverS >= 0 {
          if *recoverS < 5 {
              fmt.Fprintln(os.Stderr, "--recover-s must be >= 5")
              return 2
          }
          merged["shed_recover_seconds"] = *recoverS
          anyFlag = true
      }
      if !anyFlag {
          fmt.Fprintln(os.Stderr, "at least one threshold flag required")
          return 2
      }

      buf, err := json.Marshal(merged)
      if err != nil {
          fmt.Fprintf(os.Stderr, "marshal circuit_config: %v\n", err)
          return 1
      }

      // Use existing UpdateUpstreamAdmin query (from Phase 3 pattern) — if it
      // accepts full config JSONB, pass buf; otherwise use a dedicated

      params := gen.UpdateUpstreamAdminParams{

          CircuitConfig: buf,

          // existing values. Adapt to actual sqlc params struct.
      }

          fmt.Fprintf(os.Stderr, "update: %v\n", err)
          return 1
      }
      fmt.Fprintf(os.Stdout, "thresholds updated: %s => %s\n", *upstream, string(buf))
      return 0
  }
  ```

  ```go
  package main

  import (
      "context"
      "flag"

      "github.com/jackc/pgx/v5/pgtype"

      "github.com/ifixtelecom/gpu-ifix/gateway/internal/db/gen"

  // runTenantSetShedLimits implements `gatewayctl tenant set-shed-limits`.
  // Called from runTenant dispatcher in tenant.go when args[0] == "set-shed-limits".
  func runTenantSetShedLimits(ctx context.Context, args []string, log *slog.Logger) int {
      fs := flag.NewFlagSet("tenant set-shed-limits", flag.ContinueOnError)
      slug := fs.String("slug", "", "tenant slug (required)")
      llm := fs.Int("llm", -1, "local_inflight_max_llm; -1 = unchanged")
      stt := fs.Int("stt", -1, "local_inflight_max_stt; -1 = unchanged")
      embed := fs.Int("embed", -1, "local_inflight_max_embed; -1 = unchanged")
      tier := fs.String("tier", "", "priority_tier (S|A|B); empty = unchanged")
      if err := fs.Parse(args); err != nil {
          return 2
      }
      if *slug == "" {
          fmt.Fprintln(os.Stderr, "--slug required")
          return 2
      }

      params := gen.UpdateTenantShedLimitsParams{Slug: *slug}
      anyFlag := false
      if *llm >= 0 {
          if *llm == 0 || *llm > 1000 {
              fmt.Fprintln(os.Stderr, "--llm must be in [1, 1000]")
              return 2
          }
          params.LocalInflightMaxLlm = pgtype.Int4{Int32: int32(*llm), Valid: true}
          anyFlag = true
      }
      if *stt >= 0 {
          if *stt == 0 || *stt > 1000 {
              fmt.Fprintln(os.Stderr, "--stt must be in [1, 1000]")
              return 2
          }
          params.LocalInflightMaxStt = pgtype.Int4{Int32: int32(*stt), Valid: true}
          anyFlag = true
      }
      if *embed >= 0 {
          if *embed == 0 || *embed > 1000 {
              fmt.Fprintln(os.Stderr, "--embed must be in [1, 1000]")
              return 2
          }
          params.LocalInflightMaxEmbed = pgtype.Int4{Int32: int32(*embed), Valid: true}
          anyFlag = true
      }
      if *tier != "" {
          if *tier != "S" && *tier != "A" && *tier != "B" {
              fmt.Fprintln(os.Stderr, "--tier must be S|A|B")
              return 2
          }
          params.PriorityTier = pgtype.Text{String: *tier, Valid: true}
          anyFlag = true
      }
      if !anyFlag {
          fmt.Fprintln(os.Stderr, "at least one flag required (--llm, --stt, --embed, --tier)")
          return 2
      }

      pool := mustPool(log)
      defer pool.Close()
      q := gen.New(pool)
      if err := q.UpdateTenantShedLimits(ctx, params); err != nil {
          fmt.Fprintf(os.Stderr, "update: %v\n", err)
          return 1
      }
      fmt.Fprintf(os.Stdout, "tenant %s updated\n", *slug)
      return 0
  }
  ```
