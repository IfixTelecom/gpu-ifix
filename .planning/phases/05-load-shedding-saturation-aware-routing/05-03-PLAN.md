  ```go
  // Package shed (latency.go): lockless ring buffer for per-upstream request
  // latency samples. Read pattern is once-per-tick by the FSM (see fsm.go);
  // write pattern is once-per-response by the middleware (see middleware.go).
  //
  // Race-benign-by-design: two concurrent writers to the same slot lose one
  // sample each out of several thousand. CONTEXT.md D-A2 / RESEARCH Pitfall 2
  // accept this explicitly.
  package shed

  import (
      "sort"
      "sync/atomic"
  )

  // LatencyRing holds up to `size` latest latency samples (ms) per upstream.
  // Size is configured via SHED_LATENCY_RING_SIZE (default 200).
  type LatencyRing struct {
      buf  []uint32
      size uint64
      idx  atomic.Uint64 // monotonic write index; (idx-1)%size gives current slot
  }

  // NewLatencyRing pre-allocates the buffer. Callers must NOT mutate size later.
  func NewLatencyRing(size int) *LatencyRing {
      if size <= 0 {
          size = 200
      }
      return &LatencyRing{buf: make([]uint32, size), size: uint64(size)}
  }

  // Record stores a sample. Amortized O(1), lockless. Race benign on slot
  // collision (see package doc).
  func (r *LatencyRing) Record(ms uint32) {
      if r == nil {
          return
      }
      i := r.idx.Add(1) - 1
      r.buf[i%r.size] = ms
  }

  // P95 returns the 95th percentile of currently-stored samples.
  // Returns 0 if no samples have been written.
  func (r *LatencyRing) P95() uint32 {
      if r == nil {
          return 0
      }
      snap := make([]uint32, r.size)
      copy(snap, r.buf)
      out := snap[:0]
      for _, v := range snap {
          if v > 0 {
              out = append(out, v)
          }
      }
      if len(out) == 0 {
          return 0
      }
      sort.Slice(out, func(i, j int) bool { return out[i] < out[j] })
      idx := int(float64(len(out))*0.95 + 0.5)
      if idx >= len(out) {
          idx = len(out) - 1
      }
      return out[idx]
  }
  ```

  ```go
  func TestLatencyRing_EmptyReturnsZero(t *testing.T) {
      r := NewLatencyRing(100)
      if got := r.P95(); got != 0 {
          t.Fatalf("empty P95 = %d, want 0", got)

  func TestLatencyRing_UniformSamplesP95(t *testing.T) {
      r := NewLatencyRing(200)
      for i := uint32(1); i <= 100; i++ {
          r.Record(i)
      }
      p95 := r.P95()
      // 95th percentile of uniform [1..100] with ceil-round at 0.95*100+0.5=95.5 -> idx 95 -> value 96
      // sort.Slice idx rounds via int(0.95*100+0.5)=95, so out[95]=96
      if p95 < 94 || p95 > 96 {
          t.Fatalf("P95 of [1..100] = %d, expected ~95", p95)
      }
  }

  func TestLatencyRing_OverwritesOldSamples(t *testing.T) {
      r := NewLatencyRing(10)
      for i := uint32(1); i <= 20; i++ {
          r.Record(i)
      }
      // After 20 writes into size-10 ring, only [11..20] survive
      p95 := r.P95()
      if p95 < 19 || p95 > 20 {
          t.Fatalf("P95 of last-10 after 20 writes = %d, expected ~20", p95)
      }
  }

  func TestLatencyRing_ConcurrentWrites(t *testing.T) {
      // Race detection: this test is expected to flag benign races on slot
      // writes under -race. Acceptable per D-A2. We verify no panic/no bad
      // final state.
      r := NewLatencyRing(512)
      var wg sync.WaitGroup
      for g := 0; g < 10; g++ {
          wg.Add(1)
          go func(id int) {
              defer wg.Done()
              for i := 0; i < 1000; i++ {
                  r.Record(uint32(i + id*1000))
              }
          }(g)
      }
      wg.Wait()
      if p := r.P95(); p == 0 {
          t.Fatalf("P95 after concurrent writes = 0, expected >0")
      }
  }
  ```

  - `transition` chama `onChange` + `obs.GatewayShedState.WithLabelValues(upstream).Set(float64(to))` + `obs.GatewayShedTransitions.WithLabelValues(upstream, from.String(), to.String()).Inc()`.

  ```go
  package shed

  import (
      "log/slog"
      "sync/atomic"
      "time"

      "github.com/ifixtelecom/gpu-ifix/gateway/internal/obs"
  )

  type State int32

  const (
      StateOff State = iota
      StateArmed
      StateOn
      StateRecovering
  )

  func (s State) String() string {
      switch s {
      case StateOff:
          return "off"
      case StateArmed:
          return "armed"
      case StateOn:
          return "on"
      case StateRecovering:
          return "recovering"
      }
      return "unknown"
  }

  // Config holds per-upstream FSM tunables derived from upstreams.circuit_config JSONB.
  type Config struct {
      Upstream       string
      ArmSeconds     int64 // 30 default
      RecoverSeconds int64 // 60 default
  }

  // Signals carries one tick of observed values for the FSM 2-of-3 gate (D-A1).
  type Signals struct {
      InflightOverMax bool
      P95OverMax      bool
      VramOverMax     bool
      VramUnknown     bool // true when DCGM scraper is in fail-open mode (D-A3)
  }

  // FSM is the per-upstream shed state machine. Transitions are driven by
  // the global FSMTicker calling Evaluate once per second. All reads of State()
  // from the hot path are lockless.
  type FSM struct {
      upstream  string
      state     atomic.Int32
      enteredAt atomic.Int64 // Unix seconds when current state entered
      cfg       atomic.Pointer[Config]
      onChange  func(from, to State, reason string)
      log       *slog.Logger
  }

  // NewFSM creates an FSM initialised to StateOff at time "now".
  func NewFSM(upstream string, cfg Config, onChange func(from, to State, reason string), log *slog.Logger) *FSM {
      f := &FSM{
          upstream: upstream,
          onChange: onChange,
          log:      log.With("module", "SHED_FSM", "upstream", upstream),
      }
      cfg.Upstream = upstream
      f.cfg.Store(&cfg)
      f.enteredAt.Store(time.Now().Unix())
      obs.GatewayShedState.WithLabelValues(upstream).Set(float64(StateOff))
      return f
  }

  // UpdateConfig atomically swaps the config (called after NOTIFY hot-reload).
  // The FSM may already be mid-cycle; D-C5 ensures hysteresis is preserved.
  func (f *FSM) UpdateConfig(cfg Config) {
      cfg.Upstream = f.upstream
      f.cfg.Store(&cfg)
  }

  // State returns the current state (lockless hot-path read).
  func (f *FSM) State() State {
      return State(f.state.Load())
  }

  // EnteredAt returns when the current state was entered (monotonic unix seconds).
  func (f *FSM) EnteredAt() time.Time {
      return time.Unix(f.enteredAt.Load(), 0)
  }

  // Evaluate applies D-C1 transition rules given one tick's signals.
  // Expected to be called by the global FSMTicker once per second.
  func (f *FSM) Evaluate(now time.Time, sig Signals) {
      score := 0
      if sig.InflightOverMax {
          score++
      }
      if sig.P95OverMax {
          score++
      }
      if sig.VramOverMax && !sig.VramUnknown {
          score++
      }
      saturated := score >= 2

      cfg := f.cfg.Load()
      if cfg == nil {
          return
      }
      current := State(f.state.Load())
      entered := f.enteredAt.Load()
      elapsed := now.Unix() - entered

      switch current {
      case StateOff:
          if saturated {
              f.transition(StateOff, StateArmed, now, "signal_rose")
          }
      case StateArmed:
          if !saturated {
              f.transition(StateArmed, StateOff, now, "signal_dropped_during_arm")
          } else if elapsed >= cfg.ArmSeconds {
              f.transition(StateArmed, StateOn, now, "arm_timeout_sustained")
          }
      case StateOn:
          if !saturated {
              f.transition(StateOn, StateRecovering, now, "signal_dropped")
          }
      case StateRecovering:
          if saturated {
              f.transition(StateRecovering, StateOn, now, "signal_returned_during_recover")
          } else if elapsed >= cfg.RecoverSeconds {
              f.transition(StateRecovering, StateOff, now, "recover_timeout_clean")
          }
      }
  }

  // Transition exposes synthetic state changes (used by Subscribe for remote
  // events and by gatewayctl shed-force overrides).
  func (f *FSM) Transition(newState State, reason string) {
      from := State(f.state.Load())
      f.transition(from, newState, time.Now(), reason)
  }

  // transition does the compare-and-swap. If state changed under us (another
  // tick fired), skip — next tick re-evaluates fresh. Obs callbacks + onChange
  // are fired AFTER successful CAS, off the hot path.
  func (f *FSM) transition(from, to State, now time.Time, reason string) {
      if from == to {
          return
      }
      if !f.state.CompareAndSwap(int32(from), int32(to)) {
          return
      }
      f.enteredAt.Store(now.Unix())
      obs.GatewayShedState.WithLabelValues(f.upstream).Set(float64(to))
      obs.GatewayShedTransitions.WithLabelValues(f.upstream, from.String(), to.String()).Inc()
      f.log.Info("shed FSM transition",
          "from", from.String(),
          "to", to.String(),
          "reason", reason,
          "at", now.Format(time.RFC3339),
      )
      if f.onChange != nil {
          f.onChange(from, to, reason)
      }
  }
  ```

  ```go
  package shed

  import (
      "log/slog"
      "sync"

      "github.com/redis/go-redis/v9"
  )

  // Set aggregates per-upstream FSMs and the remote-state map used by the
  // subscribe loop (CONTEXT D-C3). Authoritative state is the in-process FSM;
  // Redis is a mirror never the source of truth (D-C3, same as breaker).
  type Set struct {
      rdb *redis.Client
      log *slog.Logger

      mu          sync.RWMutex
      fsms        map[string]*FSM
      remoteState map[string]State // last state reported by other replicas via Pub/Sub
      onChange    func(upstream string, from, to State, reason string)
      defaultCfg  Config
  }

  // Options configures a new Set.
  type Options struct {
      DefaultArmSeconds     int64
      DefaultRecoverSeconds int64
      OnChange              func(upstream string, from, to State, reason string)
  }

  // NewSet creates an empty Set. Call Rebuild to populate FSMs from the
  // upstream loader names list.
  func NewSet(rdb *redis.Client, log *slog.Logger, opt Options) *Set {
      if opt.DefaultArmSeconds == 0 {
          opt.DefaultArmSeconds = 30
      }
      if opt.DefaultRecoverSeconds == 0 {
          opt.DefaultRecoverSeconds = 60
      }
      return &Set{
          rdb:         rdb,
          log:         log.With("module", "SHED"),
          fsms:        make(map[string]*FSM),
          remoteState: make(map[string]State),
          onChange:    opt.OnChange,
          defaultCfg:  Config{ArmSeconds: opt.DefaultArmSeconds, RecoverSeconds: opt.DefaultRecoverSeconds},
      }
  }

  // Rebuild adds FSMs for names not yet present and removes FSMs whose names
  // no longer appear. Existing FSMs keep their state (hot-reload invariant).
  func (s *Set) Rebuild(names []string) {
      s.mu.Lock()
      defer s.mu.Unlock()
      want := make(map[string]bool, len(names))
      for _, n := range names {
          want[n] = true
      }
      for n := range s.fsms {
          if !want[n] {
              delete(s.fsms, n)
              delete(s.remoteState, n)
          }
      }
      for _, n := range names {
          if _, ok := s.fsms[n]; !ok {
              cfg := s.defaultCfg
              cfg.Upstream = n
              name := n // capture
              cb := func(from, to State, reason string) {
                  if s.onChange != nil {
                      s.onChange(name, from, to, reason)
                  }
              }
              s.fsms[n] = NewFSM(n, cfg, cb, s.log)
          }
      }
  }

  // Get returns the FSM for name or (nil, false).
  func (s *Set) Get(name string) (*FSM, bool) {
      s.mu.RLock()
      defer s.mu.RUnlock()
      f, ok := s.fsms[name]
      return f, ok
  }

  // State is a hot-path convenience: returns current state or StateOff if unknown.
  func (s *Set) State(name string) State {
      if f, ok := s.Get(name); ok {
          return f.State()
      }
      return StateOff
  }

  // ForEach iterates FSMs under RLock. Callback must NOT block (holds RLock).
  func (s *Set) ForEach(fn func(upstream string, f *FSM)) {
      s.mu.RLock()
      defer s.mu.RUnlock()
      for n, f := range s.fsms {
          fn(n, f)
      }
  }

  // Names returns the current set of managed upstream names (snapshot).
  func (s *Set) Names() []string {
      s.mu.RLock()
      defer s.mu.RUnlock()
      names := make([]string, 0, len(s.fsms))
      for n := range s.fsms {
          names = append(names, n)
      }
      return names
  }

  // ApplyRemoteEvent records a state transition reported by another replica.
  // The in-process FSM is NOT forced to that state — the remoteState map is
  // used by gatewayctl shed-state for dashboard-level observability only.
  // Convergence to the actual FSM state happens via the periodic reconcile
  // loop (see subscribe.go Pitfall 3).
  func (s *Set) ApplyRemoteEvent(upstream string, state State) {
      s.mu.Lock()
      defer s.mu.Unlock()
      s.remoteState[upstream] = state
  }

  // RemoteState returns the last state reported by any replica for upstream.
  func (s *Set) RemoteState(upstream string) (State, bool) {
      s.mu.RLock()
      defer s.mu.RUnlock()
      st, ok := s.remoteState[upstream]
      return st, ok
  }
  ```
